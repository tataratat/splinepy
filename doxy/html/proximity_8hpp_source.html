157 RealArray& Metric(const RealArray2D& spline_gradient_,
158 const RealArray& delta_guess_) {
159 for (int i{}; i < metric.size(); ++i) {
160 metric[i] = delta_guess_.InnerProduct(&spline_gradient_(i, 0));
161 }
162 return metric;
163 }
