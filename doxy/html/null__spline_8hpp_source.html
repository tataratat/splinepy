127// maxdim to pre-create null splines
128#ifdef SPLINEPY_MORE
129constexpr static int kMaxLookupDim{3};
130#else
131constexpr static int kMaxLookupDim{10};
132#endif
133using LookupArray_ =
134 std::array<std::array<std::shared_ptr<SplinepyBase>, kMaxLookupDim>,
135 kMaxLookupDim>;
137inline static const LookupArray_ kNullSplineLookup = [] {
138 LookupArray_ lookup;
139 int i{0};
140 for (auto& para_dim_array : lookup) {
141 int j{0};
142 for (auto& dim_element : para_dim_array) {
143 dim_element = std::make_shared<NullSpline>(i + 1, j + 1);
144 ++j;
145 }
146 ++i;
147 }
148 return lookup;
149}();
150
151std::shared_ptr<SplinepyBase> NullSplineFromLookup(const int para_dim,
152 const int dim) {
153 return kNullSplineLookup[para_dim - 1][dim - 1];
154}
