39template<typename ValueType>
40static bool CheckPyArrayShape(const py::array_t<ValueType> arr,
41 const std::vector<int>& shape,
42 const bool throw_ = true) {
43 const std::size_t expected_dim = shape.size();
44 if (expected_dim != static_cast<std::size_t>(arr.ndim())) {
45 if (!throw_)
46 return false;
47 splinepy::utils::PrintAndThrowError("Array dim mismatch.",
48 "Expected -",
49 expected_dim,
50 "Given -",
51 arr.ndim());
52 }
53 const py::ssize_t* arrshape = arr.shape();
54 for (std::size_t i{}; i < expected_dim; ++i) {
55 const int& shape_i = shape[i];
56 if (shape_i < 0) {
57 continue;
58 } else {
59 if (shape_i != arrshape[i]) {
60 if (!throw_)
61 return false;
62 splinepy::utils::PrintAndThrowError("Array shape mismatch",
63 "in dimension [",
64 i,
65 "].",
66 "Expected -",
67 shape_i,
68 "Given -",
69 arrshape[i]);
70 }
71 }
72 }
73 return true;
74}
