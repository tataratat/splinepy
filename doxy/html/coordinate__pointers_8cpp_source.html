29void ControlPointPointers::SetRow(const int id_, const double* values) {
30 if (invalid_) {
31 return;
32 }
33
34 // allow negative id
35 const int len = Len();
36 const int id = WrapId(id_, len);
37
38 if (for_rational_) {
39 const double& weight = *(weight_pointers_->weights_[id]);
40 double* coord = coordinate_begins_[id];
41 for (int i{}; i < Dim(); ++i) {
42 coord[i] = values[i] * weight;
43 }
44 } else {
45 double* coord = coordinate_begins_[id];
46 for (int i{}; i < Dim(); ++i) {
47 coord[i] = values[i];
48 }
49 }
50}

52void ControlPointPointers::Sync(const double* values) {
53 if (invalid_) {
54 return;
55 }
56
57 const int dim = Dim();
58
59 if (for_rational_) {
60 const auto& weights = weight_pointers_->weights_;
61
62 for (int i{}; i < Len(); ++i) {
63 // get destinations and sources
64 double* current_coord = coordinate_begins_[i];
65 const double* current_value = &values[i * dim];
66 const double& current_weight = *weights[i];
67
68 for (int j{}; j < dim; ++j) {
69 // saves weighted control points
70 current_coord[j] = current_value[j] * current_weight;
71 }
72 }
73 } else {
74 for (int i{}; i < Len(); ++i) {
75 // get destinations and sources
76 double* current_coord = coordinate_begins_[i];
77 const double* current_value = &values[i * dim];
78
79 for (int j{}; j < dim; ++j) {
80 // saves weighted control points
81 current_coord[j] = current_value[j];
82 }
83 }
84 }
85}

87void WeightPointers::SetRow(const int id_, double const& value) {
88 if (invalid_) {
89 return;
90 }
91
92 // allow negative id
93 const int len = Len();
94 const int id = WrapId(id_, len);
95
96 if (auto cpp = control_point_pointers_.lock()) {
97 // adjustment factor - new value divided by previous factor;
98 double& current_weight = *weights_[id];
99 const double adjust_factor = value / current_weight;
100
101 double* current_coordinate = cpp->coordinate_begins_[id];
102 for (int i{}; i < cpp->dim_; ++i) {
103 current_coordinate[i] *= adjust_factor;
104 }
105
106 // save new weight
107 current_weight = value;
108 } else {
109 splinepy::utils::PrintAndThrowError(
110 "Missing related control point pointers. Please help us and report "
111 "this issue to github.com/tataratat/splinepy, thank you!");
112 }
113}
