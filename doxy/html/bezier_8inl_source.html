38template<std::size_t para_dim, std::size_t dim>
39typename Bezier<para_dim, dim>::Base_
40Bezier<para_dim, dim>::CreateBase(const int* degrees,
41 const double* control_points) {
42
43 std::array<std::size_t, para_dim> bm_degrees{};
44 std::size_t ncps{1};
45
46 // formulate degrees
47 for (std::size_t i{}; i < para_dim; ++i) {
48 bm_degrees[i] = degrees[i];
49 ncps *= degrees[i] + 1;
50 }
51
52 // formulate control_points
53 std::vector<Coordinate_> bm_control_points(ncps);
54 for (std::size_t i{}; i < ncps; ++i) {
55 if constexpr (dim > 1) {
56 for (std::size_t j = 0; j < dim; j++) {
57 bm_control_points[i][j] = control_points[i * dim + j];
58 }
59 } else {
60 bm_control_points[i] = control_points[i];
61 }
62 }
63 return Base_(bm_degrees, bm_control_points);
64}

93template<std::size_t para_dim, std::size_t dim>
94void Bezier<para_dim, dim>::SplinepyCurrentProperties(
95 int* degrees,
96 std::vector<std::vector<double>>* knot_vectors /* untouched */,
97 double* control_points,
98 double* weights /* untouched */) const {
99
100 // degrees
101 if (degrees) {
102 for (std::size_t i{}; i < kParaDim; ++i) {
103 degrees[i] = static_cast<int>(Base_::GetDegrees()[i]);
104 }
105 }
106
107 // control_points
108 if (control_points) {
109 const std::size_t ncps = Base_::control_points.size();
110 for (std::size_t i{}; i < ncps; ++i) {
111 if constexpr (dim > 1) {
112 for (std::size_t j{}; j < kDim; ++j) {
113 control_points[i * kDim + j] = Base_::control_points[i][j];
114 }
115 } else {
116 control_points[i] = Base_::control_points[i];
117 }
118 }
119 }
120}

122template<std::size_t para_dim, std::size_t dim>
123std::shared_ptr<typename Bezier<para_dim, dim>::ControlPointPointers_>
124Bezier<para_dim, dim>::SplinepyControlPointPointers() {
125 if (SplinepyBase_::control_point_pointers_
126 && SplinepyBase_::control_point_pointers_->Len()
127 == SplinepyNumberOfControlPoints()) {
128 return SplinepyBase_::control_point_pointers_;
129 }
130 auto cpp = std::make_shared<ControlPointPointers_>();
131 cpp->dim_ = kDim;
132 cpp->coordinate_begins_.reserve(SplinepyNumberOfControlPoints());
133 if constexpr (kDim == 1) {
134 for (auto& control_point : Base_::control_points) {
135 cpp->coordinate_begins_.push_back(&control_point);
136 }
137 } else {
138 for (auto& control_point : Base_::control_points) {
139 cpp->coordinate_begins_.push_back(control_point.data());
140 }
141 }
142
143 SplinepyBase_::control_point_pointers_ = cpp;
144
145 return cpp;
146}

282template<std::size_t para_dim, std::size_t dim>
283std::shared_ptr<SplinepyBase> Bezier<para_dim, dim>::SplinepyMultiply(
284 const std::shared_ptr<SplinepyBase>& a) const {
285
286 SplinepySplineNameMatches(
287 *this,
288 *a,
289 "Spline multiplication requires splines of same type.",
290 true);
291 SplinepyParaDimMatches(
292 *this,
293 *a,
294 "Spline multiplication requires splines of same parametric dimension.",
295 true);
296 // check dimension if a is not a scalar spline
297 if (a->SplinepyDim() != 1) {
298 SplinepyDimMatches(
299 *this,
300 *a,
301 (std::string) "Spline multiplication requires splines of either 1 or "
302 + "same physical dimension.",
303 true);
304 }
305
306 // good to multiply
307 if (a->SplinepyDim() == 1) {
308 auto true_a = static_cast<Bezier<para_dim, 1>&>(*a);
309 return std::make_shared<Bezier<para_dim, dim>>(
310 this->Base_::operator*(true_a));
311 } else {
312 auto true_a = static_cast<Bezier<para_dim, dim>&>(*a);
313 return std::make_shared<Bezier<para_dim, 1>>(
314 this->Base_::operator*(true_a));
315 }
316}

341template<std::size_t para_dim, std::size_t dim>
342template<bool is_rational, size_t inner_para_dim>
343std::vector<std::shared_ptr<SplinepyBase>>
344Bezier<para_dim, dim>::ConvertComposeToBase(
345 const std::shared_ptr<SplinepyBase>& inner_function) const {
346 if constexpr (is_rational) {
347 auto super_return = this->ComposeSensitivity(
348 static_cast<RationalBezier<inner_para_dim, para_dim>&>(
349 *inner_function));
350
351 std::vector<std::shared_ptr<SplinepyBase>> to_return(super_return.size());
352
353 for (size_t i{}; i < super_return.size(); ++i) {
354 to_return[i] =
355 std::make_shared<RationalBezier<inner_para_dim, 1>>(super_return[i]);
356 }
357 return to_return;
358 } else {
359 auto super_return = this->ComposeSensitivity(
360 static_cast<Bezier<inner_para_dim, para_dim>&>(*inner_function));
361
362 std::vector<std::shared_ptr<SplinepyBase>> to_return(super_return.size());
363
364 for (size_t i{}; i < super_return.size(); ++i) {
365 to_return[i] =
366 std::make_shared<Bezier<inner_para_dim, 1>>(super_return[i]);
367 }
368 return to_return;
369 }
370}

577template<std::size_t para_dim, std::size_t dim>
578std::vector<std::shared_ptr<SplinepyBase>>
579Bezier<para_dim, dim>::SplinepySplit(const int& p_dim,
580 const double& location) const {

583 Base_::SplitAtPosition(location, static_cast<std::size_t>(p_dim));
584
585 // make it splinepybase
586 std::vector<std::shared_ptr<SplinepyBase>> split;

591 }
592

630template<std::size_t para_dim, std::size_t dim>
631std::shared_ptr<SplinepyBase>
632Bezier<para_dim, dim>::SplinepyCompositionDerivative(
633 const std::shared_ptr<SplinepyBase>& inner,
634 const std::shared_ptr<SplinepyBase>& inner_derivative) const {
635 // type check
636 if (inner->SplinepySplineName().find("Bezier") == std::string::npos) {
637 splinepy::utils::PrintAndThrowError(
638 "Bezier composition derivative requires inner function to be",
639 "a bezier type. Given inner function -",
640 inner->SplinepyWhatAmI());
641 }
642
643 // composable?
644 if (inner->SplinepyDim() != this->SplinepyParaDim()) {
645 splinepy::utils::PrintAndThrowError(
646 "Spline composition requires inner function to have same physical",
647 "dimension as outer function's parametric dimension.",
648 "Outer Function:",
649 this->SplinepyWhatAmI(),
650 "/",
651 "Inner Function:",
652 inner->SplinepyWhatAmI());
653 }
654 std::array<int, para_dim> order_query;
655 order_query.fill(0);
656 order_query[0] = 1;
657 std::shared_ptr<SplinepyBase> composition_derivative =
658 this->SplinepyDerivativeSpline(order_query.data())
659 ->SplinepyCompose(inner)
660 ->SplinepyMultiply(inner_derivative->SplinepyExtractDim(0));
661
662 for (std::size_t i{1}; i < para_dim; ++i) {
663 order_query.fill(0);
664 order_query[i] = 1;
665 composition_derivative = composition_derivative->SplinepyAdd(
666 this->SplinepyDerivativeSpline(order_query.data())
667 ->SplinepyCompose(inner)
668 ->SplinepyMultiply(inner_derivative->SplinepyExtractDim(i)));
669 }
670 return composition_derivative;
671}
