36void init_knot_vector(py::module_& m) {
37 using KnotVector = bsplinelib::parameter_spaces::KnotVector;
38 using IterType = typename KnotVector::Knots_::iterator;
39 using KnotType = typename KnotVector::Knots_::value_type;
40 using DiffType = typename KnotVector::Knots_::difference_type;
41 using SizeType = typename KnotVector::Knots_::size_type;
42
43 // implementations adapted from pybind11/stl_bind.h
44
46 auto wrap_id = [](DiffType i, const SizeType n) {
47 if (i < 0) {
48 i += n;
49 }
50 if (i < 0 || (SizeType) i >= n) {
51 throw py::index_error();
52 }
53 return i;
54 };

110 if (!slice.compute(kv_size, &start, &stop, &step, &slicelength)) {
