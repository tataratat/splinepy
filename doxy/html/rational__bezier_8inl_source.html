34template<std::size_t para_dim, std::size_t dim>
35typename RationalBezier<para_dim, dim>::Base_
36RationalBezier<para_dim, dim>::CreateBase(const int* degrees,
37 const double* control_points,
38 const double* weights) {
39
40 std::array<std::size_t, para_dim> bm_degrees{};
41 std::size_t ncps{1};
42
43 // formulate degrees
44 for (std::size_t i{}; i < para_dim; ++i) {
45 bm_degrees[i] = degrees[i];
46 ncps *= degrees[i] + 1;
47 }
48
49 // formulate control_points and weights. bezman will apply weights
50 std::vector<Coordinate_> bm_control_points(ncps);
51 std::vector<Weight_> bm_weights(ncps);
52 for (std::size_t i{}; i < ncps; ++i) {
53 // weights
54 bm_weights[i] = weights[i];
55 // weighted cps
56 if constexpr (dim > 1) {
57 for (std::size_t j = 0; j < dim; j++) {
58 bm_control_points[i][j] = control_points[i * dim + j];
59 }
60 } else {
61 bm_control_points[i] = control_points[i];
62 }
63 }
64 return Base_(bm_degrees, bm_control_points, bm_weights);
65}

96template<std::size_t para_dim, std::size_t dim>
97void RationalBezier<para_dim, dim>::SplinepyCurrentProperties(
98 int* degrees,
99 std::vector<std::vector<double>>* knot_vectors /* untouched */,
100 double* control_points,
101 double* weights) const {
102
103 // degrees
104 if (degrees) {
105 for (std::size_t i{}; i < kParaDim; ++i) {
106 degrees[i] = static_cast<int>(Base_::GetDegrees()[i]);
107 }
108 }
109
110 // control_points and weights
111 if (control_points || weights) {
112 const std::size_t ncps = Base_::GetWeightedControlPoints().size();
113 for (std::size_t i{}; i < ncps; ++i) {
114 const double w = Base_::GetWeights()[i];
115 if (weights) {
116 weights[i] = w;
117 }
118 if (control_points) {
119 double inv_weight = static_cast<double>(1.) / w;
120 if constexpr (dim > 1) {
121 for (std::size_t j{}; j < kDim; ++j) {
122 control_points[i * kDim + j] =
123 Base_::GetWeightedControlPoints()[i][j] * inv_weight;
124 }
125 } else {
126 control_points[i] = Base_::GetWeightedControlPoints()[i] * inv_weight;
127 }
128 }
129 }
130 }
131}

133template<std::size_t para_dim, std::size_t dim>
134std::shared_ptr<
135 typename RationalBezier<para_dim, dim>::WeightedControlPointPointers_>
136RationalBezier<para_dim, dim>::SplinepyWeightedControlPointPointers() {
137 if (SplinepyBase_::control_point_pointers_
138 && SplinepyBase_::control_point_pointers_->Len()
139 == SplinepyNumberOfControlPoints()) {
140 return SplinepyBase_::control_point_pointers_;
141 }
142
143 // create weighted cps
144 auto wcpp = std::make_shared<WeightedControlPointPointers_>();
145 wcpp->dim_ = kDim;
146 wcpp->for_rational_ = kIsRational;
147 wcpp->coordinate_begins_.reserve(SplinepyNumberOfControlPoints());
148 if constexpr (kDim == 1) {
149 for (auto& w_control_point : Base_::GetWeightedControlPoints()) {
150 wcpp->coordinate_begins_.push_back(&w_control_point);
151 }
152 } else {
153 for (auto& w_control_point : Base_::GetWeightedControlPoints()) {
154 wcpp->coordinate_begins_.push_back(w_control_point.data());
155 }
156 }
157
158 // create weights
159 auto w = std::make_shared<WeightPointers_>();
160 w->weights_.reserve(SplinepyNumberOfControlPoints());
161 for (auto& weight : Base_::GetWeights()) {
162 w->weights_.push_back(&weight);
163 }
164
165 // reference each other
166 w->control_point_pointers_ = wcpp; // weak_ptr
167 wcpp->weight_pointers_ = w;
168
169 SplinepyBase_::control_point_pointers_ = wcpp;
170
171 return wcpp;
172}

316template<std::size_t para_dim, std::size_t dim>
317std::shared_ptr<SplinepyBase> RationalBezier<para_dim, dim>::SplinepyMultiply(
318 const std::shared_ptr<SplinepyBase>& a) const {
319
320 SplinepySplineNameMatches(
321 *this,
322 *a,
323 "Spline multiplication requires splines of same type.",
324 true);
325 SplinepyParaDimMatches(
326 *this,
327 *a,
328 "Spline multiplication requires splines of same parametric dimension.",
329 true);
330 // check dimension if a is not a scalar spline
331 if (a->SplinepyDim() != 1) {
332 SplinepyDimMatches(
333 *this,
334 *a,
335 (std::string) "Spline multiplication requires splines of either 1 or "
336 + "same physical dimension.",
337 true);
338 }
339
340 // good to multiply
341 if (a->SplinepyDim() == 1) {
342 auto true_a = static_cast<RationalBezier<para_dim, 1>&>(*a);
343 return std::make_shared<RationalBezier<para_dim, dim>>(
344 this->Base_::operator*(true_a));
345 } else {
346 auto true_a = static_cast<RationalBezier<para_dim, dim>&>(*a);
347 return std::make_shared<RationalBezier<para_dim, 1>>(
348 this->Base_::operator*(true_a));
349 }

378RationalBezier<para_dim, dim>::ConvertComposeToBase(

394 static_cast<Bezier<inner_para_dim, para_dim>&>(*inner_function));
395
396 std::vector<std::shared_ptr<SplinepyBase>> to_return(super_return.size());
397
398 for (size_t i{}; i < super_return.size(); ++i) {

612template<std::size_t para_dim, std::size_t dim>
613std::vector<std::shared_ptr<SplinepyBase>>
614RationalBezier<para_dim, dim>::SplinepySplit(const int& p_dim,
615 const double& location) const {
616 // split
617 auto bm_splitted =
618 Base_::SplitAtPosition(location, static_cast<std::size_t>(p_dim));
619
620 // make it splinepybase
621 std::vector<std::shared_ptr<SplinepyBase>> split;
622 const std::size_t n_splitted = bm_splitted.size();
623 split.reserve(n_splitted); // this should be always 2
624 for (std::size_t i{}; i < n_splitted; ++i) {
625 split.emplace_back(

667template<std::size_t para_dim, std::size_t dim>
668std::shared_ptr<SplinepyBase>
669RationalBezier<para_dim, dim>::SplinepyCompositionDerivative(
670 const std::shared_ptr<SplinepyBase>& inner,
671 const std::shared_ptr<SplinepyBase>& inner_derivative) const {
672 // type check
673 if (inner->SplinepySplineName().find("Bezier") == std::string::npos) {
674 splinepy::utils::PrintAndThrowError(
675 "Bezier composition derivative requires inner function to be",
676 "a bezier type. Given inner function -",
677 inner->SplinepyWhatAmI());
678 }
679
680 // composable?
681 if (inner->SplinepyDim() != this->SplinepyParaDim()) {
682 splinepy::utils::PrintAndThrowError(
683 "Spline composition requires inner function to have same physical",
684 "dimension as outer function's parametric dimension.",
685 "Outer Function:",
686 this->SplinepyWhatAmI(),
687 "/",
688 "Inner Function:",
689 inner->SplinepyWhatAmI());
690 }
691 std::array<int, para_dim> order_query;
692 order_query.fill(0);
693 order_query[0] = 1;
694 std::shared_ptr<SplinepyBase> composition_derivative =
695 this->SplinepyDerivativeSpline(order_query.data())
696 ->SplinepyCompose(inner)
697 ->SplinepyMultiply(inner_derivative->SplinepyExtractDim(0));
698
699 for (std::size_t i{1}; i < para_dim; ++i) {
700 order_query.fill(0);
701 order_query[i] = 1;
702 composition_derivative = composition_derivative->SplinepyAdd(
703 this->SplinepyDerivativeSpline(order_query.data())
704 ->SplinepyCompose(inner)
705 ->SplinepyMultiply(inner_derivative->SplinepyExtractDim(i)));
706 }
707 return composition_derivative;
708}
