48template<typename PointersType, bool same_sized_values>
49void SetRows(PointersType& pointers,
50 const py::array_t<int> ids,
51 const py::array_t<double>& values) {
52 const int n_rows = ids.size();
53
54 if constexpr (same_sized_values) {
55 if (pointers.Len() * pointers.Dim() != values.size()) {
56 splinepy::utils::PrintAndThrowError("Size mismatch. Expecting",
57 pointers.Len() * pointers.Dim(),
58 "but, values are ",
59 values.size());
60 }
61 } else {
62 if (n_rows * pointers.Dim() != values.size()) {
63 splinepy::utils::PrintAndThrowError("Size mismatch. Expecting",
64 pointers.Dim() * ids.size(),
65 "but, values are ",
66 values.size());
67 }
68 }
69
70 if (pointers.Dim() != values.shape(1)) {
71 splinepy::utils::PrintAndThrowError("Dimension mismatch. Expecting",
72 pointers.Dim(),
73 "but values are",
74 values.shape(1));
75 }
76 pointers.template SetRows<same_sized_values>(
77 static_cast<int*>(ids.request().ptr),
78 n_rows,
79 static_cast<double*>(values.request().ptr));
80}

82template<typename PointersType>
83void Sync(PointersType& pointers, const py::array_t<double>& values) {
84 if (pointers.Len() * pointers.Dim() != values.size()) {
85 splinepy::utils::PrintAndThrowError("Size mismatch, Expecting",
86 pointers.Len() * pointers.Dim(),
87 "but values are",
88 values.size());
89 }
90 pointers.Sync(static_cast<double*>(values.request().ptr));
91}

99void init_coordinate_pointers(py::module& m) {
100 using ControlPointPointers = splinepy::utils::ControlPointPointers;
101 using WeightPointers = splinepy::utils::WeightPointers;
102
103 py::class_<ControlPointPointers, std::shared_ptr<ControlPointPointers>>
104 klasse(m, "ControlPointPointers");
105 klasse.def("len", &Len<ControlPointPointers>)
106 .def("dim", &Dim<ControlPointPointers>)
107 .def(
108 "set_row",
109 [](ControlPointPointers& pointers,
110 const int id,
111 const py::array_t<double>& values) {
112 if (pointers.Dim() != values.size()) {
113 splinepy::utils::PrintAndThrowError("Size mismatch. Expecting",
114 pointers.dim_,
115 "but, values are ",
116 values.size(),
117 ".");
118 }
119 pointers.SetRow(id, static_cast<double*>(values.request().ptr));
120 },
121 py::arg("id"),
122 py::arg("values"))
123 .def("set_rows",
124 &SetRows<ControlPointPointers, false>,
125 py::arg("id"),
126 py::arg("values"))
127 .def("sync_rows",
128 &SetRows<ControlPointPointers, true>,
129 py::arg("id"),
130 py::arg("values"))
131 .def("sync", &Sync<ControlPointPointers>, py::arg("values"));
132
133 py::class_<WeightPointers, std::shared_ptr<WeightPointers>> klasse_weight(
134 m,
135 "WeightPointers");
136 klasse_weight.def("len", &Len<WeightPointers>)
137 .def("dim", &Dim<WeightPointers>)
138 .def("set_row", &WeightPointers::SetRow, py::arg("id"), py::arg("values"))
139 .def("set_rows",
140 &SetRows<WeightPointers, false>,
141 py::arg("id"),
142 py::arg("values"))
143 .def("sync_rows",
144 &SetRows<WeightPointers, true>,
145 py::arg("id"),
146 py::arg("values"))
147 .def("sync", &Sync<WeightPointers>, py::arg("values"));
148}
