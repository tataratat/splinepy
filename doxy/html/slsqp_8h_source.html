38/* optimizer * */
39/* *********************************************************************** */
40/* SLSQP S EQUENTIAL L EAST SQ UARES P ROGRAMMING */
41/* TO SOLVE GENERAL NONLINEAR OPTIMIZATION PROBLEMS */
42/* *********************************************************************** */
43/* * * */
44/* * * */
45/* * A NONLINEAR PROGRAMMING METHOD WITH * */
46/* * QUADRATIC PROGRAMMING SUBPROBLEMS * */
47/* * * */
48/* * * */
49/* * THIS SUBROUTINE SOLVES THE GENERAL NONLINEAR PROGRAMMING PROBLEM * */
50/* * * */
51/* * MINIMIZE F(X) * */
52/* * * */
53/* * SUBJECT TO C (X) .EQ. 0 , J = 1,...,MEQ * */
54/* * J * */
55/* * * */
56/* * C (X) .GE. 0 , J = MEQ+1,...,M * */
57/* * J * */
58/* * * */
59/* * XL .LE. X .LE. XU , I = 1,...,N. * */
60/* * I I I * */
61/* * * */
62/* * THE ALGORITHM IMPLEMENTS THE METHOD OF HAN AND POWELL * */
63/* * WITH BFGS-UPDATE OF THE B-MATRIX AND L1-TEST FUNCTION * */
64/* * WITHIN THE STEPLENGTH ALGORITHM. * */
65/* * * */
66/* * PARAMETER DESCRIPTION: * */
67/* * ( * MEANS THIS PARAMETER WILL BE CHANGED DURING CALCULATION ) * */
68/* * * */
69/* * M IS THE TOTAL NUMBER OF CONSTRAINTS, M .GE. 0 * */
70/* * MEQ IS THE NUMBER OF EQUALITY CONSTRAINTS, MEQ .GE. 0 * */
71/* * LA SEE A, LA .GE. MAX(M,1) * */
72/* * N IS THE NUMBER OF VARIABLES, N .GE. 1 * */
73/* * * X() X() STORES THE CURRENT ITERATE OF THE N VECTOR X * */
74/* * ON ENTRY X() MUST BE INITIALIZED. ON EXIT X() * */
75/* * STORES THE SOLUTION VECTOR X IF MODE = 0. * */
76/* * XL() XL() STORES AN N VECTOR OF LOWER BOUNDS XL TO X. * */
77/* * ELEMENTS MAY BE NAN TO INDICATE NO LOWER BOUND. * */
78/* * XU() XU() STORES AN N VECTOR OF UPPER BOUNDS XU TO X. * */
79/* * ELEMENTS MAY BE NAN TO INDICATE NO UPPER BOUND. * */
80/* * F IS THE VALUE OF THE OBJECTIVE FUNCTION. * */
81/* * C() C() STORES THE M VECTOR C OF CONSTRAINTS, * */
82/* * EQUALITY CONSTRAINTS (IF ANY) FIRST. * */
83/* * DIMENSION OF C MUST BE GREATER OR EQUAL LA, * */
84/* * which must be GREATER OR EQUAL MAX(1,M). * */
85/* * G() G() STORES THE N VECTOR G OF PARTIALS OF THE * */
86/* * OBJECTIVE FUNCTION; DIMENSION OF G MUST BE * */
87/* * GREATER OR EQUAL N+1. * */
88/* * A(),LA,M,N THE LA BY N + 1 ARRAY A() STORES * */
89/* * THE M BY N MATRIX A OF CONSTRAINT NORMALS. * */
90/* * A() HAS FIRST DIMENSIONING PARAMETER LA, * */
91/* * WHICH MUST BE GREATER OR EQUAL MAX(1,M). * */
92/* * F,C,G,A MUST ALL BE SET BY THE USER BEFORE EACH CALL. * */
93/* * * ACC ABS(ACC) CONTROLS THE FINAL ACCURACY. * */
94/* * IF ACC .LT. ZERO AN EXACT LINESEARCH IS PERFORMED,* */
95/* * OTHERWISE AN ARMIJO-TYPE LINESEARCH IS USED. * */
96/* * * ITER PRESCRIBES THE MAXIMUM NUMBER OF ITERATIONS. * */
97/* * ON EXIT ITER INDICATES THE NUMBER OF ITERATIONS. * */
98/* * * MODE MODE CONTROLS CALCULATION: * */
99/* * REVERSE COMMUNICATION IS USED IN THE SENSE THAT * */
100/* * THE PROGRAM IS INITIALIZED BY MODE = 0; THEN IT IS* */
101/* * TO BE CALLED REPEATEDLY BY THE USER UNTIL A RETURN* */
102/* * WITH MODE .NE. IABS(1) TAKES PLACE. * */
103/* * IF MODE = -1 GRADIENTS HAVE TO BE CALCULATED, * */
104/* * WHILE WITH MODE = 1 FUNCTIONS HAVE TO BE CALCULATED */
105/* * MODE MUST NOT BE CHANGED BETWEEN SUBSEQUENT CALLS * */
106/* * OF SQP. * */
107/* * EVALUATION MODES: * */
108/* * MODE = -1: GRADIENT EVALUATION, (G&A) * */
109/* * 0: ON ENTRY: INITIALIZATION, (F,G,C&A) * */
110/* * ON EXIT : REQUIRED ACCURACY FOR SOLUTION OBTAINED * */
111/* * 1: FUNCTION EVALUATION, (F&C) * */
112/* * * */
113/* * FAILURE MODES: * */
114/* * 2: NUMBER OF EQUALITY CONSTRAINTS LARGER THAN N * */
115/* * 3: MORE THAN 3*N ITERATIONS IN LSQ SUBPROBLEM * */
116/* * 4: INEQUALITY CONSTRAINTS INCOMPATIBLE * */
117/* * 5: SINGULAR MATRIX E IN LSQ SUBPROBLEM * */
118/* * 6: SINGULAR MATRIX C IN LSQ SUBPROBLEM * */
119/* * 7: RANK-DEFICIENT EQUALITY CONSTRAINT SUBPROBLEM HFTI* */
120/* * 8: POSITIVE DIRECTIONAL DERIVATIVE FOR LINESEARCH * */
121/* * 9: MORE THAN ITER ITERATIONS IN SQP * */
122/* * >=10: WORKING SPACE W OR JW TOO SMALL, * */
123/* * W SHOULD BE ENLARGED TO L_W=MODE/1000 * */
124/* * JW SHOULD BE ENLARGED TO L_JW=MODE-1000*L_W * */
125/* * * W(), L_W W() IS A ONE DIMENSIONAL WORKING SPACE, * */
126/* * THE LENGTH L_W OF WHICH SHOULD BE AT LEAST * */
127/* * (3*N1+M)*(N1+1) for LSQ * */
128/* * +(N1-MEQ+1)*(MINEQ+2) + 2*MINEQ for LSI * */
129/* * +(N1+MINEQ)*(N1-MEQ) + 2*MEQ + N1 for LSEI * */
130/* * + N1*N/2 + 2*M + 3*N + 3*N1 + 1 for SLSQPB * */
131/* * with MINEQ = M - MEQ + 2*N1 & N1 = N+1 * */
132/* * NOTICE: FOR PROPER DIMENSIONING OF W IT IS RECOMMENDED TO * */
133/* * COPY THE FOLLOWING STATEMENTS INTO THE HEAD OF * */
134/* * THE CALLING PROGRAM (AND REMOVE THE COMMENT C) * */
135/* ####################################################################### */
136/* INTEGER LEN_W, LEN_JW, M, N, N1, MEQ, MINEQ */
137/* PARAMETER (M=... , MEQ=... , N=... ) */
138/* PARAMETER (N1= N+1, MINEQ= M-MEQ+N1+N1) */
139/* PARAMETER (LEN_W= */
140/* $ (3*N1+M)*(N1+1) */
141/* $ +(N1-MEQ+1)*(MINEQ+2) + 2*MINEQ */
142/* $ +(N1+MINEQ)*(N1-MEQ) + 2*MEQ + N1 */
143/* $ +(N+1)*N/2 + 2*M + 3*N + 3*N1 + 1, */
144/* $ LEN_JW=MINEQ) */
145/* DOUBLE PRECISION W(LEN_W) */
146/* INTEGER JW(LEN_JW) */
147/* ####################################################################### */
148/* * THE FIRST M+N+N*N1/2 ELEMENTS OF W MUST NOT BE * */
149/* * CHANGED BETWEEN SUBSEQUENT CALLS OF SLSQP. * */
150/* * ON RETURN W(1) ... W(M) CONTAIN THE MULTIPLIERS * */
151/* * ASSOCIATED WITH THE GENERAL CONSTRAINTS, WHILE * */
152/* * W(M+1) ... W(M+N(N+1)/2) STORE THE CHOLESKY FACTOR* */
153/* * L*D*L(T) OF THE APPROXIMATE HESSIAN OF THE * */
154/* * LAGRANGIAN COLUMNWISE DENSE AS LOWER TRIANGULAR * */
155/* * UNIT MATRIX L WITH D IN ITS 'DIAGONAL' and * */

159/* * THE SEARCH DIRECTION TO THE SOLUTION X* * */
160/* * * JW(), L_JW JW() IS A ONE DIMENSIONAL INTEGER WORKING SPACE * */
161/* * THE LENGTH L_JW OF WHICH SHOULD BE AT LEAST * */
162/* * MINEQ * */

306/* MINIMIZE with respect to X */
307/* ||E*X - F|| */
308/* 1/2 T */
309/* WITH UPPER TRIANGULAR MATRIX E = +D *L , */
310/* -1/2 -1 */
311/* AND VECTOR F = -D *L *G, */
312/* WHERE THE UNIT LOWER TRIDIANGULAR MATRIX L IS STORED COLUMNWISE */
313/* DENSE IN THE N*(N+1)/2 ARRAY L WITH VECTOR D STORED IN ITS */
314/* 'DIAGONAL' THUS SUBSTITUTING THE ONE-ELEMENTS OF L */
315/* SUBJECT TO */
316/* A(J)*X - B(J) = 0 , J=1,...,MEQ, */
317/* A(J)*X - B(J) >=0, J=MEQ+1,...,M, */
318/* XL(I) <= X(I) <= XU(I), I=1,...,N, */
319/* ON ENTRY, THE USER HAS TO PROVIDE THE ARRAYS L, G, A, B, XL, XU. */
320/* WITH DIMENSIONS: L(N*(N+1)/2), G(N), A(LA,N), B(M), XL(N), XU(N) */
321/* THE WORKING ARRAY W MUST HAVE AT LEAST THE FOLLOWING DIMENSION: */
322/* DIM(W) = (3*N+M)*(N+1) for LSQ */
323/* +(N-MEQ+1)*(MINEQ+2) + 2*MINEQ for LSI */
324/* +(N+MINEQ)*(N-MEQ) + 2*MEQ + N for LSEI */
325/* with MINEQ = M - MEQ + 2*N */
326/* ON RETURN, NO ARRAY WILL BE CHANGED BY THE SUBROUTINE. */
327/* X STORES THE N-DIMENSIONAL SOLUTION VECTOR */
328/* Y STORES THE VECTOR OF LAGRANGE MULTIPLIERS OF DIMENSION */
329/* M+N+N (CONSTRAINTS+LOWER+UPPER BOUNDS) */
330/* MODE IS A SUCCESS-FAILURE FLAG WITH THE FOLLOWING MEANINGS: */
331/* MODE=1: SUCCESSFUL COMPUTATION */
332/* 2: ERROR RETURN BECAUSE OF WRONG DIMENSIONS (N<1) */
333/* 3: ITERATION COUNT EXCEEDED BY NNLS */
334/* 4: INEQUALITY CONSTRAINTS INCOMPATIBLE */
335/* 5: MATRIX E IS NOT OF FULL RANK */
336/* 6: MATRIX C IS NOT OF FULL RANK */
337/* 7: RANK DEFECT IN HFTI */

357/* FOR MODE=1, THE SUBROUTINE RETURNS THE SOLUTION X OF */
358/* EQUALITY & INEQUALITY CONSTRAINED LEAST SQUARES PROBLEM LSEI : */
359/* MIN ||E*X - F|| */
360/* X */
361/* S.T. C*X = D, */
362/* G*X >= H. */
363/* USING QR DECOMPOSITION & ORTHOGONAL BASIS OF NULLSPACE OF C */
364/* CHAPTER 23.6 OF LAWSON & HANSON: SOLVING LEAST SQUARES PROBLEMS. */
365/* THE FOLLOWING DIMENSIONS OF THE ARRAYS DEFINING THE PROBLEM */
366/* ARE NECESSARY */
367/* DIM(E) : FORMAL (LE,N), ACTUAL (ME,N) */
368/* DIM(F) : FORMAL (LE ), ACTUAL (ME ) */
369/* DIM(C) : FORMAL (LC,N), ACTUAL (MC,N) */
370/* DIM(D) : FORMAL (LC ), ACTUAL (MC ) */
371/* DIM(G) : FORMAL (LG,N), ACTUAL (MG,N) */
372/* DIM(H) : FORMAL (LG ), ACTUAL (MG ) */
373/* DIM(X) : FORMAL (N ), ACTUAL (N ) */
374/* DIM(W) : 2*MC+ME+(ME+MG)*(N-MC) for LSEI */
375/* +(N-MC+1)*(MG+2)+2*MG for LSI */
376/* DIM(JW): MAX(MG,L) */

457/* T */
458/* MINIMIZE 1/2 X X SUBJECT TO G * X >= H. */
459/* C.L. LAWSON, R.J. HANSON: 'SOLVING LEAST SQUARES PROBLEMS' */
460/* PRENTICE HALL, ENGLEWOOD CLIFFS, NEW JERSEY, 1974. */
461/* PARAMETER DESCRIPTION: */
462/* G(),MG,M,N ON ENTRY G() STORES THE M BY N MATRIX OF */
463/* LINEAR INEQUALITY CONSTRAINTS. G() HAS FIRST */
464/* DIMENSIONING PARAMETER MG */
465/* H() ON ENTRY H() STORES THE M VECTOR H REPRESENTING */
466/* THE RIGHT SIDE OF THE INEQUALITY SYSTEM */
467/* REMARK: G(),H() WILL NOT BE CHANGED DURING CALCULATIONS BY LDP */
468/* X() ON ENTRY X() NEED NOT BE INITIALIZED. */
469/* ON EXIT X() STORES THE SOLUTION VECTOR X IF MODE=1. */
470/* XNORM ON EXIT XNORM STORES THE EUCLIDEAN NORM OF THE */
471/* SOLUTION VECTOR IF COMPUTATION IS SUCCESSFUL */
472/* W() W IS A ONE DIMENSIONAL WORKING SPACE, THE LENGTH */
473/* OF WHICH SHOULD BE AT LEAST (M+2)*(N+1) + 2*M */
474/* ON EXIT W() STORES THE LAGRANGE MULTIPLIERS */
475/* ASSOCIATED WITH THE CONSTRAINTS */
476/* AT THE SOLUTION OF PROBLEM LDP */
477/* INDEX() INDEX() IS A ONE DIMENSIONAL INTEGER WORKING SPACE */
478/* OF LENGTH AT LEAST M */
