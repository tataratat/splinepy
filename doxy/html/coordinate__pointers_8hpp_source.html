36template<typename IndexType, typename SizeType>
37IndexType WrapId(IndexType i, const SizeType n) {
38 if (i < 0) {
39 i += n;

42#ifndef NDEBUG
43 if (i < 0 || (SizeType) i >= n) {
44 splinepy::utils::PrintAndThrowError("Index out of range");

48 return i;

150template<bool same_sized_values>
151void ControlPointPointers::SetRows(const int* ids,
152 const int n_rows,
153 const double* values) {
154 if (invalid_) {
155 return;
156 }
157 const auto dim = Dim();
158 const auto len = Len();
159
160 if (for_rational_) {
161 const auto& weights = weight_pointers_->weights_;
162 for (int i{}; i < n_rows; ++i) {
163 // get id
164 const auto current_id = WrapId(ids[i], len);
165 // get destinations and sources
166 auto* current_coord = coordinate_begins_[current_id];
167 const double* current_value;
168 if constexpr (same_sized_values) {
169 current_value = &values[current_id * dim];
170 } else {
171 current_value = &values[i * dim];
172 }
173 const auto& current_weight = *weights[current_id];
174
175 for (int j{}; j < dim; ++j) {
176 // saves weighted control points
177 current_coord[j] = current_value[j] * current_weight;
178 }
179 }
180 } else {
181 for (int i{}; i < n_rows; ++i) {
182 // get id
183 const auto current_id = WrapId(ids[i], len);
184 // get destinations and sources
185 auto* current_coord = coordinate_begins_[current_id];
186 const double* current_value;
187 if constexpr (same_sized_values) {
188 current_value = &values[current_id * dim];
189 } else {
190 current_value = &values[i * dim];
191 }
192
193 for (int j{}; j < dim; ++j) {
194 // saves weighted control points
195 current_coord[j] = current_value[j];
196 }
197 }
198 }
199}
