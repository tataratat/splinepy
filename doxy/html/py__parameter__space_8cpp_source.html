38void init_parameter_space(py::module_& m) {
39 using PSpace = bsplinelib::parameter_spaces::ParameterSpaceBase;
40 using KV = bsplinelib::parameter_spaces::KnotVector;
41 using KVPtr = std::shared_ptr<KV>;
42
43 // implementations adapted from pybind11/stl_bind.h
45 auto wrap_id = [](int i, int n) -> int {
46 if (i < 0) {
47 i += n;
48 }
49 if (i < 0 || i >= n) {
50 throw py::index_error();
51 }
52 return i;
53 };
54
55 auto to_list = [](const PSpace& p) -> py::list {
56 // if this were to be a list, this would mean a shallow copy.
57 // however, within splinepy's implementation, copy() means deepcopy.
58 // so, deepcopy it is.
59 py::list kvs(p.ParaDim());
60 for (int i{}; i < p.ParaDim(); ++i) {
61 KVPtr kv = p.GetKnotVector(i);
62 py::array_t<double> out_kv(kv->GetSize());
63 std::copy_n(kv->GetKnots().data(),
64 kv->GetKnots().size(),
65 static_cast<double*>(out_kv.request().ptr));
66 kvs[i] = out_kv;
67 }
68 return kvs;
69 };
70
71 py::class_<PSpace, std::shared_ptr<PSpace>> klasse(m, "ParameterSpace");
72
73 klasse
74 .def(
75 "__len__",
76 [](const PSpace& p) { return p.ParaDim(); },
77 "Returns number of parameter dimension.")
78
79 .def(
80 "__iter__",
81 [](PSpace& p) {
82 return py::make_iterator<
83 py::return_value_policy::reference_internal,
84 KVPtr*,
85 KVPtr*,
86 KVPtr&>(p.KnotVectorsBegin(), p.KnotVectorsEnd());
87 },
88 py::keep_alive<0, 1>(),
89 "Support iterations of element references.")
90
91 .def(
92 "__getitem__",
93 [wrap_id](PSpace& p, int i) -> KVPtr& {
94 i = wrap_id(i, p.ParaDim());
95 return p.GetKnotVector(i);
96 },
97 py::return_value_policy::reference_internal,
98 "int based __getitem__, which returns reference.")
99 .def(
100 "__getitem__",
101 [](const PSpace& p, const py::slice& slice) -> py::list {
102 std::size_t start{}, stop{}, step{}, slicelength{};
103 if (!slice.compute(static_cast<std::size_t>(p.ParaDim()),
104 &start,
105 &stop,
106 &step,
107 &slicelength)) {
108 throw py::error_already_set();
109 }
110 py::list items{};
111 for (std::size_t i{}; i < slicelength; ++i) {
112 items.append(p.GetKnotVector(static_cast<int>(start)));
113 start += step;
114 }
115 return items;
116 },
117 py::arg("slice"),
118 "support slice based __getitem__.")
119 .def(
120 "__setitem__",
121 [wrap_id](PSpace& p, int i, const KVPtr new_kv) {
122 i = wrap_id(i, p.ParaDim());
123 auto& p_kv = p.GetKnotVector(i);
124 if (p_kv->GetSize() != new_kv->GetSize()) {
125 splinepy::utils::PrintAndThrowError(
126 "Size mismatch of lhs & rhs knot vectors");
127 }
128 // update is simple assignment - note that this is reference
129 // this is a same behavir as list
130 p_kv = new_kv;
131 },
132 "Single knot vector assignment with another knot vector.")
133 .def(
134 "__setitem__",
135 [wrap_id](PSpace& p, int i, const py::array_t<double>& new_kv) {
136 i = wrap_id(i, p.ParaDim());
137 auto& p_kv = p.GetKnotVector(i);
138 const int kv_size = p_kv->GetSize();
139 if (kv_size != new_kv.size()) {
140 splinepy::utils::PrintAndThrowError(
141 "Size mismatch of lhs & rhs knot vectors");
142 }
143 // update is value copy
144 double* p_data = p_kv->GetKnots().data();
145 std::copy_n(static_cast<double*>(new_kv.request().ptr),
146 kv_size,
147 p_data);
148
149 // sanity check
150 p_kv->ThrowIfTooSmallOrNotNonDecreasing();
151 },
152 "Single knot vector assignment with an array")
153 .def("__add__",
154 [to_list](const PSpace& p, py::list& next) {
155 return to_list(p) + next;
156 })
157 .def("__radd__",
158 [to_list](const PSpace& p, py::list& next) {
159 return next + to_list(p);
160 })
161 .def("__repr__",
162 [](const PSpace& p) {
163 std::string s{"ParameterSpace ["};
164 const int para_dim = p.ParaDim();
165 const int last{para_dim - 1};
166 for (int i{}; i < para_dim; ++i) {
167 s.append(p.GetKnotVector(i)->StringRepresentation());
168 if (i != last) {
169 s.append(", ");
170 }
171 }
172 s.append("]");
173 return s;
174 })
175 .def("copy", [to_list](const PSpace& p) { return to_list(p); })
176 .def("unique_knots", [](const PSpace& p) {
177 py::list unique_knots;
178
179 for (int i{}; i < p.ParaDim(); ++i) {
180 const int degree = p.GetDegree(i);
181 const auto& kv = p.GetKnotVector(i);
182 const auto& knots = kv->GetKnots();
183
184 // create array
185 // this uses `new` to allocate memory and we can use this with
186 // py::capsule to avoid copy or resize at the end.
187 // See: github.com/pybind/pybind11/issues/1042#issuecomment-325941022
188 splinepy::utils::Array<double> u_knots(static_cast<int>(knots.size())
189 - (2 * degree));
190
191 // create a capsule to release data once it's done
192 auto capsule = py::capsule(u_knots.data(), [](void* data) {
193 delete reinterpret_cast<double*>(data);
194 });
195
196 // unlike KnotVector::UniqueKnots, this checks within
197 // valid regions.
198 int n_uk{1};
199 std::unique_copy(
200 knots.begin() + degree,
201 knots.end() - degree,
202 u_knots.data(),
203 [&n_uk](const double& lhs_knot, const double& rhs_knot) {
204 if (std::abs(lhs_knot - rhs_knot)
205 < bsplinelib::parameter_spaces::kEpsilon) {
206 return true;
207 } else {
208 ++n_uk;
209 return false;
210 };
211 });
212
213 unique_knots.append(
214 py::array_t<double>(n_uk, u_knots.data(), capsule));
215
216 // mark ownership to false.
217 // we can do it in append(), but if that line fails, then it'd
218 // be a memory leak.
219 u_knots.TransferOwnership();
220 }
221
222 return unique_knots;
223 });
224}
