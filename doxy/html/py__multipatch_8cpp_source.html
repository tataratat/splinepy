59ToCoreSplineVector(py::list pysplines,

62 const int nthreads) {
63 // prepare return obj
64 const int n_splines = static_cast<int>(pysplines.size());
65 std::vector<PySpline::CoreSpline_> core_splines(n_splines);

68 for (int i{begin}; i < end; ++i) {

79 core_splines[i] =
80 spline.template cast<std::shared_ptr<PySpline>>()->Core();
81 }
82 };
83 splinepy::utils::NThreadExecution(to_core, n_splines, nthreads);
84
85 return core_splines;

103template<typename T>
104IntVector ArgSort(const std::vector<T>& v) {
105 IntVector idx(v.size());
106 std::iota(idx.begin(), idx.end(), 0);
107 std::stable_sort(idx.begin(), idx.end(), [&v](int const& i1, int const& i2) {
108 return v[i1] < v[i2];
109 });
110 return idx;
111}

113void RaiseMismatch(const CoreSplineVector& splist,
114 const std::string name,
115 const int para_dim,
116 const int dim,
117 const IntVector& degrees,
118 const IntVector& control_mesh_resolutions,
119 const int nthreads) {
120 // for verbose output
121 std::unordered_map<std::string, IntVectorVector> mismatches{};
122
123 // check flags
124 bool check_name{}, check_para_dim{}, check_dim{}, check_degrees{},
125 check_control_mesh_resolutions{};
126 // input vector sizes
127 const int d_size = degrees.size();
128 const int cmr_size = control_mesh_resolutions.size();
129 // true para_dim to use as reference -> will be updated / compared
130 int ref_para_dim{para_dim};
131
132 // parse input and allocate vector for mismatch book keeping
133 if (name.size() > 0) {
134 check_name = true;
135 mismatches["name"].resize(nthreads);
136 }
137 if (dim > 0) {
138 check_dim = true;
139 mismatches["dim"].resize(nthreads);
140 }
141 if (d_size > 0) {
142 check_degrees = true;
143 ref_para_dim = d_size;
144 mismatches["degrees"].resize(nthreads);
145 }
146 if (cmr_size > 0) {
147 check_control_mesh_resolutions = true;
148 mismatches["control_mesh_resolutions"].resize(nthreads);
149 }
150 if (para_dim > 0 || check_degrees || check_control_mesh_resolutions) {
151 check_para_dim = true;
152
153 // sanity check
154 for (const auto& candidate : {para_dim, d_size, cmr_size}) {
155 if (candidate != ref_para_dim && candidate > 0) {
156 splinepy::utils::PrintAndThrowError(
157 "Mismatch in given para_dim (",
158 para_dim,
159 "), size of degrees, (",
160 d_size,
161 "), size of control_mesh_resolutions (",
162 cmr_size,
163 ").");
164 }
165 }
166 mismatches["para_dim"].resize(nthreads);
167 }
168
169 // lambda for nthread comparison
170 auto check_mismatch_step = [&](int, int, const int i_thread) {
171 // alloc vectors in case we need to compare
172 IntVector spline_degree(ref_para_dim), spline_cmr(ref_para_dim);
173 int total_ = static_cast<int>(splist.size());
174 for (int i{i_thread}; i < total_; i += nthreads) {
175 // get spline to check
176 const auto& spline = splist[i];
177
178 // skip null splines
179 if (spline->SplinepyIsNull()) {
180 continue;
181 }
182
183 // default value for para_dim match
184 bool para_dim_matches{true};
185
186 // check
187 if (check_name && (name != spline->SplinepySplineName())) {
188 mismatches["name"][i_thread].push_back(i);
189 }
190 if (check_para_dim && (ref_para_dim != spline->SplinepyParaDim())) {
191 mismatches["para_dim"][i_thread].push_back(i);
192 para_dim_matches = false;
193 }
194 if (check_dim && (dim != spline->SplinepyDim())) {
195 mismatches["dim"][i_thread].push_back(i);
196 }
197 // check properties that are relevant iff para_dim matches
198 if (para_dim_matches) {
199 if (check_degrees) {
200 spline->SplinepyCurrentProperties(spline_degree.data(),
201 nullptr,
202 nullptr,
203 nullptr);
204 if (spline_degree != degrees) {
205 mismatches["degrees"][i_thread].push_back(i);
206 }
207 }
208 if (check_control_mesh_resolutions) {
209 spline->SplinepyControlMeshResolutions(spline_cmr.data());
210 if (spline_cmr != control_mesh_resolutions) {
211 mismatches["control_mesh_resolutions"][i_thread].push_back(i);
212 }
213 }
214 }
215 }
216 };
217
218 splinepy::utils::NThreadExecution(check_mismatch_step,
219 static_cast<int>(splist.size()),
220 nthreads);
221
222 // prepare output or maybe exit
223 std::unordered_map<std::string, IntVector> concat_mismatches{};
224 bool raise{false};
225
226 for (const auto& [key, mismatch_per_threads] : mismatches) {
227 auto& concat_vec = concat_mismatches[key];
228 for (const auto& mismatch : mismatch_per_threads) {
229 const auto m_size = mismatch.size();
230 if (m_size != 0) {
231 concat_vec.insert(concat_vec.end(), mismatch.begin(), mismatch.end());
232 }
233 }
234 if (concat_vec.size() != 0) {
235 raise = true;
236 }
237 };
238
239 // everything matches
240 if (!raise) {
241 return;
242 }
243
244 // form mismatch info
245 std::string mismatch_info{};
246 for (const auto& [key, concat_mismatch] : concat_mismatches) {
247 mismatch_info += "\n[" + key + "] : ";
248 for (const auto& ids : concat_mismatch) {
249 mismatch_info += std::to_string(ids) + ", ";
250 }
251 }
252
253 splinepy::utils::PrintAndThrowError("Found mismatches.", mismatch_info);
254}

256void RaiseMismatch(const CoreSplineVector& splist0,
257 const CoreSplineVector& splist1,
258 const bool name,
259 const bool dimensionality,
260 const int field_dim,
261 const bool degrees,
262 const bool control_mesh_resolutions,
263 const int nthreads) {
264 // len check
265 if (splist0.size() != splist1.size()) {
266 splinepy::utils::PrintAndThrowError(
267 "Size mismatch between first and second list of splines.",
268 splist0.size(),
269 "vs",
270 splist1.size());
271 }
272 // for verbose output
273 std::unordered_map<std::string, IntVectorVector> mismatches{};
274 // parse input and allocate vector for mismatch book keeping
275 if (name) {
276 mismatches["name"].resize(nthreads);
277 }
278 if (dimensionality) {
279 mismatches["dim"].resize(nthreads);
280 }
281 if (degrees) {
282 mismatches["degrees"].resize(nthreads);
283 }
284 if (control_mesh_resolutions) {
285 mismatches["control_mesh_resolutions"].resize(nthreads);
286 }
287 if (dimensionality || degrees || control_mesh_resolutions) {
288 mismatches["para_dim"].resize(nthreads);
289 }
290
291 // lambda for nthread comparison
292 auto check_mismatch_step = [&](int, int, const int i_thread) {
293 // alloc some tmp vector if needed
294 IntVector int_vec0, int_vec1;
295 const int total_ = static_cast<int>(splist0.size());
296
297 for (int i{i_thread}; i < total_; i += nthreads) {
298 // get spline to check
299 const auto& spline0 = splist0[i];
300 const auto& spline1 = splist1[i];
301
302 // default value for para_dim match
303 bool para_dim_matches{true};
304
305 // check dims
306 if (dimensionality
307 && (spline0->SplinepyParaDim() != spline1->SplinepyParaDim())) {
308 mismatches["para_dim"][i_thread].push_back(i);
309 para_dim_matches = false;
310 }
311 if (dimensionality && (field_dim != spline1->SplinepyDim())) {
312 mismatches["dim"][i_thread].push_back(i);
313 }
314 // null spline should have at least same dims. but nothing else matters.
315 // skip null splines
316 if (spline1->SplinepyIsNull() || spline0->SplinepyIsNull()) {
317 continue;
318 }
319
320 if (name
321 && (spline0->SplinepySplineName() != spline1->SplinepySplineName())) {
322 mismatches["name"][i_thread].push_back(i);
323 }
324
325 // check properties that are relevant iff para_dim matches
326 if (para_dim_matches && (degrees || control_mesh_resolutions)) {
327 // alloc some space
328 int_vec0.resize(spline0->SplinepyParaDim());
329 int_vec1.resize(spline1->SplinepyParaDim());
330
331 if (degrees) {
332 spline0->SplinepyCurrentProperties(int_vec0.data(),
333 nullptr,
334 nullptr,
335 nullptr);
336 spline1->SplinepyCurrentProperties(int_vec1.data(),
337 nullptr,
338 nullptr,
339 nullptr);
340
341 if (int_vec0 != int_vec1) {
342 mismatches["degrees"][i_thread].push_back(i);
343 }
344 }
345
346 if (control_mesh_resolutions) {
347 spline0->SplinepyControlMeshResolutions(int_vec0.data());
348 spline1->SplinepyControlMeshResolutions(int_vec1.data());
349
350 if (int_vec0 != int_vec1) {
351 mismatches["control_mesh_resolutions"][i_thread].push_back(i);
352 }
353 }
354 }
355 }
356 };
357
358 splinepy::utils::NThreadExecution(check_mismatch_step,
359 static_cast<int>(splist0.size()),
360 nthreads);
361
362 // prepare output or maybe exit
363 std::unordered_map<std::string, IntVector> concat_mismatches{};
364 bool raise{false};
365
366 for (const auto& [key, mismatch_per_threads] : mismatches) {
367 auto& concat_vec = concat_mismatches[key];
368 for (const auto& mismatch : mismatch_per_threads) {
369 const auto m_size = mismatch.size();
370 if (m_size != 0) {
371 concat_vec.insert(concat_vec.end(), mismatch.begin(), mismatch.end());
372 }
373 }
374 if (concat_vec.size() != 0) {
375 raise = true;
376 }
377 };
378
379 // everything matches
380 if (!raise) {
381 return;
382 }
383
384 // form mismatch info
385 std::string mismatch_info{};
386 for (const auto& [key, concat_mismatch] : concat_mismatches) {
387 mismatch_info += "\n[" + key + "] : ";
388 for (const auto& ids : concat_mismatch) {
389 mismatch_info += std::to_string(ids) + ", ";
390 }
391 }
392
393 splinepy::utils::PrintAndThrowError("Found mismatches.", mismatch_info);
394}

396py::array_t<int>
397FindConnectivityFromCenters(const py::array_t<double>& face_center_vertices,
398 const int parametric_dimension,
399 const py::array_t<double>& metric,
400 const double tolerance) {
401
402 // -- Auxiliary data --
403 const int number_of_element_faces = parametric_dimension * 2;
404 const int number_of_patches =
405 face_center_vertices.shape(0) / number_of_element_faces;
406 const int physical_dimension = face_center_vertices.shape(1);
407 const int number_of_center_vertices{
408 static_cast<int>(face_center_vertices.shape(0))};
409 const double tolerance_squared{tolerance * tolerance};
410 const double* metric_ptr = static_cast<double*>(metric.request().ptr);
411 const double* face_center_vertices_ptr =
412 static_cast<double*>(face_center_vertices.request().ptr);
413
414 // Consistency check
415 if (!(face_center_vertices.shape(0) % number_of_element_faces == 0)) {
416 splinepy::utils::PrintAndThrowError(
417 "Number of corner vertices invalid. Must be a multiple of the number "
418 "of vertices per patch");
419 }
420 if (!(metric.size() == face_center_vertices.shape(1))) {
421 splinepy::utils::PrintAndThrowError(
422 "Incompatible size for metric. Must match physical dimension of face "
423 "center vertices");
424 }
425 if (!(number_of_center_vertices % number_of_element_faces == 0)) {
426 splinepy::utils::PrintAndThrowError(
427 "Inconsistent number of Center vertices. Must be divisible by "
428 "parametric_dimension*2");
429 }
430
431 // Init return type
432 py::array_t<int> connectivity(number_of_patches * number_of_element_faces);
433 int* connectivity_ptr = static_cast<int*>(connectivity.request().ptr);
434 // Init connectivity and metric value
435 // (-1 : boundary, -2 : untouched)
436 std::fill(connectivity_ptr,
437 connectivity_ptr + number_of_patches * number_of_element_faces,
438 -2);
439
440 // Assure Metric is normed and non-zero
441 const std::vector<double>& normed_metric = [&metric_ptr,
442 &physical_dimension]() {
443 double euclidian_norm = 0.;
444 for (int i_phys{}; i_phys < physical_dimension; i_phys++) {
445 euclidian_norm += metric_ptr[i_phys] * metric_ptr[i_phys];
446 }
447 std::vector<double> metric(physical_dimension, 1.);
448 euclidian_norm = std::sqrt(euclidian_norm);
449 if (euclidian_norm < 1e-20) {
450 return metric;
451 } else {
452 const double inv_euclidian_norm = 1 / euclidian_norm;
453 for (int i_phys{}; i_phys < physical_dimension; i_phys++) {
454 metric[i_phys] = metric_ptr[i_phys] * inv_euclidian_norm;
455 }
456 }
457 return metric;
458 }();
459
460 // Auxiliary function to determine distance between two points in face_center
461 auto squared_euclidian_distance =
462 [&face_center_vertices_ptr,
463 &physical_dimension](const int& i_start, const int& i_end) -> double {
464 double squared_euclidian_distance_{};
465 for (int i_phys{}; i_phys < physical_dimension; i_phys++) {
466 const double distance_c =
467 face_center_vertices_ptr[i_end * physical_dimension + i_phys]
468 - face_center_vertices_ptr[i_start * physical_dimension + i_phys];
469 squared_euclidian_distance_ += distance_c * distance_c;
470 }
471 return squared_euclidian_distance_;
472 };
473
474 std::vector<double> scalar_metric{};
475 scalar_metric.reserve(number_of_center_vertices);
476
477 // Check Metric Dimension and Vector Size
478 for (int i_vertex{}; i_vertex < number_of_center_vertices; i_vertex++) {
479 double metric_v = normed_metric[0]
480 * face_center_vertices_ptr[i_vertex * physical_dimension];
481 for (int j_phys{1}; j_phys < physical_dimension; j_phys++) {
482 metric_v =
483 normed_metric[j_phys]
484 * face_center_vertices_ptr[i_vertex * physical_dimension + j_phys];
485 }
486 scalar_metric.push_back(metric_v);
487 }
488
489 // Sort Metric Vector
490 const auto metric_order_indices = ArgSort(scalar_metric);
491
492 // Loop over points
493 for (int lower_limit = 0; lower_limit < number_of_center_vertices - 1;
494 lower_limit++) {
495 // Loop over all points regardless of whether they have been touched or not,
496 // and then check the validity of the connection Point already processed
497 bool found_duplicate = false;
498 // Now check allowed range for duplicates
499 int upper_limit = lower_limit + 1;
500 while (upper_limit < number_of_center_vertices
501 && (scalar_metric[metric_order_indices[upper_limit]]
502 - scalar_metric[metric_order_indices[lower_limit]])
503 < tolerance) {
504 // Check if the two points are duplicates
505 found_duplicate =
506 squared_euclidian_distance(metric_order_indices[lower_limit],
507 metric_order_indices[upper_limit])
508 < tolerance_squared;
509 if (found_duplicate) {
510 break;
511 } else {
512 upper_limit++;
513 }
514 }
515
516 // Now we have to check if the connection is valid
517 // 1. If another connection is found, that means, that the point it connects
518 // to has a higher index in the metric tensor. If the current point does
519 // already have a neighbor, that means that more than one point connect
520 // -> Error
521 if (found_duplicate) {
522 if (connectivity_ptr[metric_order_indices[lower_limit]] != -2) {
523 splinepy::utils::PrintAndThrowError(
524 "Found conflicting interceptions, where more than two points are "
525 "in the same position. Expected -2 got",
526 connectivity_ptr[metric_order_indices[lower_limit]]);
527 }
528
529 // If both tests passed, update connectivity
530 connectivity_ptr[metric_order_indices[lower_limit]] =
531 static_cast<int>(metric_order_indices[upper_limit]);
532 connectivity_ptr[metric_order_indices[upper_limit]] =
533 static_cast<int>(metric_order_indices[lower_limit]);
534 } else {
535 // set Boundary-ID
536 if (connectivity_ptr[metric_order_indices[lower_limit]] == -2) {
537 connectivity_ptr[metric_order_indices[lower_limit]] = -1;
538 }
539 }
540 }
541
542 // Treat last remaining point in scalar metric vector
543 if (connectivity_ptr[metric_order_indices[number_of_center_vertices - 1]]
544 == -2) {
545 connectivity_ptr[metric_order_indices[number_of_center_vertices - 1]] = -1;
546 }
547
548 // Resize buffer
549 connectivity.resize({number_of_patches, number_of_element_faces});
550 return connectivity;
551}

553py::array_t<int>
554InterfacesFromBoundaryCenters(const py::array_t<double>& py_center_vertices,
555 const double& tolerance,
556 const int& parametric_dimension) {
557 // Determine Metric
558 const int physical_dimension = py_center_vertices.shape(1);
559 const int number_of_corner_points = py_center_vertices.shape(0);
560 py::array_t<double> maxvertex(physical_dimension),
561 minvertex{physical_dimension};
562
563 // Get pointers
564 double* maxvertex_ptr = static_cast<double*>(maxvertex.request().ptr);
565 double* minvertex_ptr = static_cast<double*>(minvertex.request().ptr);
566 const double* py_center_vertices_ptr =
567 static_cast<double*>(py_center_vertices.request().ptr);
568
569 // Calculate limits of center vertices
570 for (int i_dim{}; i_dim < physical_dimension; i_dim++) {
571 minvertex_ptr[i_dim] = py_center_vertices_ptr[i_dim];
572 maxvertex_ptr[i_dim] = py_center_vertices_ptr[i_dim];
573 }
574 for (int i_c{1}; i_c < number_of_corner_points; i_c++) {
575 for (int i_dim{}; i_dim < physical_dimension; i_dim++) {
576 maxvertex_ptr[i_dim] =
577 std::max(py_center_vertices_ptr[i_c * physical_dimension + i_dim],
578 maxvertex_ptr[i_dim]);
579 minvertex_ptr[i_dim] =
580 std::min(py_center_vertices_ptr[i_c * physical_dimension + i_dim],
581 minvertex_ptr[i_dim]);
582 }
583 }
584
585 // Determine Interfaces
586 return FindConnectivityFromCenters(py_center_vertices,
587 parametric_dimension,
588 maxvertex - minvertex,
589 tolerance);
590}

592void GetInterfaceOrientation(
593 const std::shared_ptr<splinepy::splines::SplinepyBase>& pyspline_start,
594 const int& boundary_start,
595 const std::shared_ptr<splinepy::splines::SplinepyBase>& pyspline_end,
596 const int& boundary_end,
597 const double& tolerance,
598 int* mappings_ptr,
599 int* orientations_ptr) {
600 // Init return values and get auxiliary data
601 const int& para_dim_ = pyspline_start->SplinepyParaDim();
602 const int& dim_ = pyspline_start->SplinepyDim();
603
604 // Checks
605 if (para_dim_ < 2) {
606 splinepy::utils::PrintAndThrowError(
607 "Curves are not supported for automatic computatation of interface "
608 "orientations");
609 }
610 if (dim_ < 2) {
611 splinepy::utils::PrintAndThrowError(
612 "Scalar splines are not supported for automatic computation of "
613 "interface orientations");
614 }
615 if ((para_dim_ != pyspline_end->SplinepyParaDim())
616 || (dim_ != pyspline_end->SplinepyDim())) {
617 splinepy::utils::PrintAndThrowError(
618 "Spline Orientation can not be checked, as they have mismatching "
619 "dimensionality. Current spline has dimensions ",
620 para_dim_,
621 "D -> ",
622 dim_,
623 "D.");
624 }
625
626 // First Check the orientation of the first entry by comparing their ids
627 const int boundary_start_p_dim = static_cast<int>(boundary_start / 2);
628 const bool boundary_start_orientation = (boundary_start % 2) == 0;
629 const int boundary_end_p_dim = static_cast<int>(boundary_end / 2);
630 const bool boundary_end_orientation = (boundary_end % 2) == 0;
631 mappings_ptr[boundary_start_p_dim] = boundary_end_p_dim;
632 // Note: Here might be a discrepancy with gismo's orientation, and it needs to
633 // be checked in the future. I am awaiting a response from gismo developers,
634 // it is poosible the orientation of the interface edge might be flipped
635 // (bugfix: negate the following expression)
636 orientations_ptr[boundary_start_p_dim] =
637 (boundary_start_orientation ^ boundary_end_orientation) ? 1 : 0;
638
640 // Calculate Parametric bounds
641 std::vector<double> bounds_start(para_dim_ * 2);
642 pyspline_start->SplinepyParametricBounds(bounds_start.data());
643 std::vector<double> bounds_end(para_dim_ * 2);
644 pyspline_end->SplinepyParametricBounds(bounds_end.data());
645 // Determine face center position in parametric space
646 std::vector<double> boundary_center_start(para_dim_),
647 boundary_center_end(para_dim_);
648
649 for (int i{}; i < para_dim_; i++) {
650 if (i == boundary_start_p_dim) {
651 boundary_center_start[i] = boundary_start_orientation
652 ? bounds_start[i]
653 : bounds_start[i + para_dim_];
654 } else {
655 boundary_center_start[i] =
656 .5 * (bounds_start[i] + bounds_start[i + para_dim_]);
657 }
658 if (i == boundary_end_p_dim) {
659 boundary_center_end[i] =
660 boundary_end_orientation ? bounds_end[i] : bounds_end[i + para_dim_];
661 } else {
662 boundary_center_end[i] = .5 * (bounds_end[i] + bounds_end[i + para_dim_]);
663 }
664 }
665
666 // Calculate Jacobians
667 std::vector<double> jacobian_start(para_dim_ * dim_),
668 jacobian_end(para_dim_ * dim_);
669 pyspline_start->SplinepyJacobian(boundary_center_start.data(),
670 jacobian_start.data());
671 pyspline_end->SplinepyJacobian(boundary_center_end.data(),
672 jacobian_end.data());
673
674 // Check the angle between the jacobian entries
675 for (int i_pd{}; i_pd < para_dim_; i_pd++) {
676 if (i_pd == boundary_start_p_dim) {
677 continue;
678 }
679 double norm_s{};
680 for (int k{}; k < dim_; k++) {
681 // [i_query * pdim * dim + i_paradim * dim + i_dim]
682 norm_s += jacobian_start[i_pd + k * para_dim_]
683 * jacobian_start[i_pd + k * para_dim_];
684 }
685 for (int j{}; j < para_dim_; j++) {
686 double norm_e{}, dot_p{};
687 for (int k{}; k < dim_; k++) {
688 dot_p += jacobian_start[i_pd + k * para_dim_]
689 * jacobian_end[j + k * para_dim_];
690 norm_e +=
691 jacobian_end[j + k * para_dim_] * jacobian_end[j + k * para_dim_];
692 }
693
694 // Check angle
695 const double cos_angle = abs(dot_p / std::sqrt(norm_s * norm_e));
696 if (cos_angle > (1. - tolerance)) {
697 mappings_ptr[i_pd] = j;
698 orientations_ptr[i_pd] = (dot_p > 0) ? 1 : 0;
699 break;
700 }
701 }
702 }

741py::array_t<int> PyMultipatch::GetInterfaceOrientations(const double tolerance,
742 const int n_threads) {
743
744 // Check if already exists
745 if (interface_orientations_.size() != 0) {
746 return interface_orientations_;
747 }
748
749 int n_patches = patches_.size();
750
751 // get para dim of the first patch
752 int para_dim = ParaDim();
753
754 const int n_faces_per_element = 2 * para_dim;
755 // 4 includes - patch id start, face id start, patch id end, face id end
756 const int n_entries_per_line = 4 + n_faces_per_element;
757
758 // Make sure Interfaces are available
759 const int* interfaces_ptr =
760 static_cast<int*>(GetInterfaces(false).request().ptr);
761
762 // Loop over interfaces to compute number of interfaces (This is generally not
763 // time critical as comparisons are much more expensive)
764 int number_of_interfaces{};
765 for (int i_interface{}; i_interface < interfaces_.size(); i_interface++) {
766 if (interfaces_ptr[i_interface] >= 0) {
767 // interface found
768 if (interfaces_ptr[i_interface] > i_interface) {
769 number_of_interfaces++;
770 }
771 }
772 }
773
774 interface_orientations_ =
775 py::array_t<int>({number_of_interfaces, n_entries_per_line});
776 int* orientations_ptr =
777 static_cast<int*>(interface_orientations_.request().ptr);
778
779 // Second loop to fill the interface information
780 int i_connection{};
781 for (int i_interface{}; i_interface < interfaces_.size(); i_interface++) {
782 if (interfaces_ptr[i_interface] >= 0) {
783 // interface found
784 if (interfaces_ptr[i_interface] > i_interface) {
785 orientations_ptr[i_connection * n_entries_per_line + 0] =
786 i_interface / n_faces_per_element;
787 orientations_ptr[i_connection * n_entries_per_line + 1] =
788 i_interface % n_faces_per_element; // std::div
789 orientations_ptr[i_connection * n_entries_per_line + 2] =
790 interfaces_ptr[i_interface] / n_faces_per_element;
791 orientations_ptr[i_connection * n_entries_per_line + 3] =
792 interfaces_ptr[i_interface] % n_faces_per_element; // std::div
793 i_connection++;
794 }
795 }
796 }
797
798 // Sanity check
799 assert(i_connection == number_of_interfaces);
800
801 const auto cpp_spline_list = ToCoreSplineVector(patches_, n_threads);
802
803 // Provide lambda for multithread execution
804 auto get_orientation = [&](const int start, const int end, int) {
805 for (int i{start}; i < end; ++i) {
806 // Assign connections (start and end faces)
807 const int& patch_id_start = orientations_ptr[i * n_entries_per_line + 0];
808 const int& face_id_start = orientations_ptr[i * n_entries_per_line + 1];
809 const int& patch_id_end = orientations_ptr[i * n_entries_per_line + 2];
810 const int& face_id_end = orientations_ptr[i * n_entries_per_line + 3];
811
812 // Determine the mappings
813 GetInterfaceOrientation(
814 cpp_spline_list[patch_id_start],
815 face_id_start,
816 cpp_spline_list[patch_id_end],
817 face_id_end,
818 tolerance,
819 &orientations_ptr[i * n_entries_per_line + 4],
820 &orientations_ptr[i * n_entries_per_line + 4 + para_dim]);
821 }
822 };
823
824 // Execute in parallel
825 splinepy::utils::NThreadExecution(get_orientation,
826 number_of_interfaces,
827 n_threads);
828
829 // Resize and return
830
831 return interface_orientations_;
832}

860
861 // This approach is a work-around for parallel execution
862 auto extract_boundaries = [&](const int start, const int end, int) {
863 // start : process-ID
864 // end : unused hence no referencing
865
866 auto& boundaries_local = lists_to_concatenate[start];
867 // Start extraction (remaining order)
868 for (int i{start}; i < end; i++) {

871 boundaries_local.append(
872 // Extract Boundary splines
873 PySpline(cpp_spline_list[i]->SplinepyExtractBoundary(j)));

875 }
876 }
877 };
878
879 // Execute in parallel
880 splinepy::utils::NThreadExecution(extract_boundaries, n_threads, n_threads);
881
882 // Concatenate list of boundaries - should only copy pointers
883 for (auto& entries : lists_to_concatenate) {
884 boundary_splines += entries;
885 }
886
887 return boundary_splines;

890int AddBoundariesFromContinuity(const py::list& boundary_splines,
891 const py::array_t<int>& boundary_interfaces,
892 py::array_t<int>& global_interfaces,
893 const double& tolerance,
894 const int& n_threads) {
895 // Check input data
896 if (static_cast<int>(py::len(boundary_splines))
897 != boundary_interfaces.shape(0)) {
898 splinepy::utils::PrintAndThrowError(
899 "Number of splines in list (",
900 py::len(boundary_splines),
901 ") and number of elements in connectivity (",
902 boundary_interfaces.shape(0),
903 ") does not match.");
904 }
905
906 // Provide auxiliary values
907 const auto cpp_spline_list = ToCoreSplineVector(boundary_splines);
908
909 const int n_boundary_patches{static_cast<int>(boundary_interfaces.shape(0))};
910 const int n_faces_per_boundary_patch{
911 static_cast<int>(boundary_interfaces.shape(1))};
912 const int para_dim_{n_faces_per_boundary_patch / 2};
913 const int dim_ = cpp_spline_list[0]->SplinepyDim();
914 const int* boundary_interfaces_ptr =
915 static_cast<int*>(boundary_interfaces.request().ptr);
916 int* global_interfaces_ptr =
917 static_cast<int*>(global_interfaces.request().ptr);
918
919 // This function is currently only implemented for 2D and 3D applications. For
920 // higher dimensions, please revise the implementation of the generalized
921 // cross product as can be found here:
922 // https://math.stackexchange.com/questions/752774/generalized-cross-product
923 if ((dim_ < 1) || (dim_ > 3)) {
924 splinepy::utils::PrintAndThrowError(
925 "G1-based boundary computation is only supported for 2D and 3D "
926 "multipatches. If this is a feature you require, please consider "
927 "writing an issue on our github");
928 }
929
930 // Auxiliary Lambdas to keep code clean
931 // Check if to normal vectors are g1 (tol > cos(phi))
932 auto areG1 = [&tolerance, &dim_](const std::vector<double>& vec0,
933 const std::vector<double>& vec1) -> bool {
934 // Checks in Debug
935 assert(static_cast<int>(vec0.size()) == dim_);
936 assert(static_cast<int>(vec1.size()) == dim_);
937
938 // Start actual computation
939 double norm0{}, norm1{}, dot_p{};
940 for (int i{}; i < dim_; i++) {
941 norm0 += vec0[i] * vec0[i];
942 norm1 += vec1[i] * vec1[i];
943 dot_p += vec0[i] * vec1[i];
944 }
945 return (tolerance > abs(1 - abs(dot_p) / std::sqrt(norm0 * norm1)));
946 };
947
948 // Tangential Vector on boundary based on its derivative
949 auto normal_vector = [&cpp_spline_list, &para_dim_, &dim_](
950 const int& patch_id,
951 const int& face_id) -> std::vector<double> {
952 // init return value (are default initialized to 0)
953 std::vector<double> para_coord(para_dim_), bounds(2 * para_dim_),
954 normal_vector(dim_), jacobian(dim_ * para_dim_);
955
956 // Auxiliary values
957 const int axis_dim = face_id / 2;
958 const int is_in_front = face_id % 2;
959
960 // Parametric Bounds
961 const auto& spline = cpp_spline_list[patch_id];
962 spline->SplinepyParametricBounds(bounds.data());
963
964 for (int i{}; i < para_dim_; i++) {
965 if (i == axis_dim) {
966 para_coord[i] = bounds[i + is_in_front * para_dim_];
967 } else {
968 para_coord[i] = .5 * (bounds[i + para_dim_] + bounds[i]);
969 }
970 }
971
972 // Compute Jacobian
973 spline->SplinepyJacobian(para_coord.data(), jacobian.data());
974
975 // Compute Cross-Product
976 if (dim_ == 2) {
977 normal_vector[0] = -jacobian[1];
978 normal_vector[1] = jacobian[0];
979 } else if (dim_ == 3) {
980 normal_vector[0] = jacobian[2] * jacobian[5] - jacobian[4] * jacobian[3];
981 normal_vector[1] = jacobian[4] * jacobian[1] - jacobian[0] * jacobian[5];
982 normal_vector[2] = jacobian[0] * jacobian[3] - jacobian[2] * jacobian[1];
983 } else {
984 // This should never happen!
985 splinepy::utils::PrintAndThrowError(
986 "G1-based boundary computation is only supported for 2D and 3D "
987 "multipatches. If this is a feature you require, please consider "
988 "writing an issue on our github");
989 }
990 return normal_vector;
991 };
992
993 // Start Computations ------------------------------------------------ //
994 // while the actual propagation needs to be performed in serial, the
995 // precomputation of interface tolerances can be performed in parallel
996 std::vector<bool> faces_are_g1(n_faces_per_boundary_patch
997 * n_boundary_patches);
998 auto precompute_tolerances = [&](const int start, const int end, int) {
999 // Loop over relevant faces
1000 for (int i{start}; i < end; i++) {
1001 // Loop over faces
1002 for (int j{}; j < n_faces_per_boundary_patch; j++) {
1003 // Interface refers to the global face id (see interface documentation)
1004 const int& connected_face_id =
1005 boundary_interfaces_ptr[i * n_faces_per_boundary_patch + j];
1006
1007 // Retrieve patch and face ID of neighbor
1008 const int adjacent_element_id =
1009 connected_face_id / n_faces_per_boundary_patch;
1010 const int adjacent_face_id =
1011 connected_face_id % n_faces_per_boundary_patch;
1012
1013 // only compute if the adjacent neighbor has higher id to prevent double
1014 // the work
1015 if (adjacent_element_id < i) {
1016 continue;
1017 }
1018 // Get tangential vector of current patch
1019 const std::vector<double> vec0 = normal_vector(i, j);
1020
1021 // Get corresponding tangential vector of neighbor patch
1022 const std::vector<double> vec1 =
1023 normal_vector(adjacent_element_id, adjacent_face_id);
1024
1025 // Check tolerance
1026 const bool is_g1 = areG1(vec0, vec1);
1027 faces_are_g1[i * n_faces_per_boundary_patch + j] = is_g1;
1028 faces_are_g1[connected_face_id] = is_g1;
1029 }
1030 }
1031 };
1032
1033 // Execute in parallel
1034 splinepy::utils::NThreadExecution(precompute_tolerances,
1035 n_boundary_patches,
1036 n_threads);
1037
1038 // std::vector can use less memory for bools
1039 std::vector<bool> is_assigned(n_boundary_patches); // defaults false
1040 std::vector<int> new_boundary_id(n_boundary_patches);
1041 std::vector<int> queued_splines{};
1042
1043 // Start Assignment
1044 // Loop over all patches
1045 int current_max_id{1};
1046 for (int i{}; i < n_boundary_patches; i++) {
1047 if (is_assigned[i]) {
1048 continue;
1049 }
1050 new_boundary_id[i] = current_max_id;
1051 is_assigned[i] = true;
1052 queued_splines.push_back(i);
1053
1054 // Start propagation
1055 while (!queued_splines.empty()) {
1056 const int current_id = queued_splines.back();
1057 queued_splines.pop_back();
1058 for (int i_face{}; i_face < n_faces_per_boundary_patch; i_face++) {
1059 const int combined_index =
1060 current_id * n_faces_per_boundary_patch + i_face;
1061 // Is the neighborface G1
1062 if (faces_are_g1[combined_index]) {
1063 // Adjacent global face id
1064 const int& adjacent_global_face_id =
1065 boundary_interfaces_ptr[combined_index];
1066 const int adjacent_patch_id =
1067 adjacent_global_face_id / n_faces_per_boundary_patch;
1068
1069 // Check if the adjacent patch is already assigned
1070 if (is_assigned[adjacent_patch_id]) {
1071 continue;
1072 } else {
1073 // Assign a BID and continue
1074 new_boundary_id[adjacent_patch_id] = current_max_id;
1075 is_assigned[adjacent_patch_id] = true;
1076 queued_splines.push_back(adjacent_patch_id);
1077 }
1078 }
1079 }
1080 }
1081
1082 // End propagation and increase id
1083 current_max_id++;
1084 }
1085
1086 // Assign the new boundary ids to the old interface-vector
1087 const int& n_interfaces = global_interfaces.size();
1088 int counter{};
1089 for (int i{}; i < n_interfaces; i++) {
1090 if (global_interfaces_ptr[i] < 0) {
1091 global_interfaces_ptr[i] = -new_boundary_id[counter];
1092 counter++;
1093 }
1094 }
1095 if (counter != n_boundary_patches) {
1096 splinepy::utils::PrintAndThrowError(
1097 counter,
1098 " new boundary ids were assigned, however ",
1099 n_boundary_patches,
1100 " were expected, which means information was lost. Abort mission");
1101 }
1102
1103 return current_max_id;
1104}

1219 // return saved
1220 if (sub_patch_centers_.size() > 0) {
1221 return sub_patch_centers_;
1222 }
1223
1224 // prepare output
1225 // from here we assume that all the splines have the same para_dim and dim
1226 const int n_splines = CorePatches().size();
1227 const int para_dim = ParaDim();
1228 const int dim = Dim();
1229 const int n_queries = 2 * para_dim;
1230 const int n_total = n_queries * n_splines;
1231 sub_patch_centers_ = py::array_t<double>({n_total, dim});
1232 double* sub_patch_centers_ptr =
1233 static_cast<double*>(sub_patch_centers_.request().ptr);
1234
1235 // pre-compute boundary centers
1236 DoubleVector para_bounds;
1237 double* para_bounds_ptr;
1238 if (!same_parametric_bounds_) {
1239 para_bounds.resize(n_total * para_dim);
1240 para_bounds_ptr = para_bounds.data();
1241 const int stride = n_queries * para_dim;
1242
1243 auto calc_para_bounds = [&](const int begin, const int end, int) {
1244 for (int i{begin}; i < end; ++i) {
1245 splinepy::splines::helpers::ScalarTypeBoundaryCenters(
1246 *core_patches_[i],
1247 &para_bounds_ptr[stride * i]);
1248 }
1249 };
1250
1251 // exe
1252 splinepy::utils::NThreadExecution(calc_para_bounds,
1253 n_splines,
1254 n_default_threads_);
1255 }
1256
1257 auto calc_sub_patch_centers_step = [&](int, int, const int i_thread) {
1258 // each thread needs one query
1259 DoubleVector queries_vector; /* unused if same_parametric_bounds=true*/
1260 double* queries;
1261
1262 // pre compute boundary centers if para bounds are the same
1263 if (same_parametric_bounds_) {
1264 queries_vector.resize(2 * para_dim * para_dim);
1265 queries = queries_vector.data();
1266 splinepy::splines::helpers::ScalarTypeBoundaryCenters(*core_patches_[0],
1267 queries);
1268 }
1269
1270 for (int i{i_thread}; i < n_total; i += n_default_threads_) {
1271 const auto [i_spline, i_query] = std::div(i, n_queries);
1272
1273 // get ptr start
1274 if (!same_parametric_bounds_) {
1275 queries = &para_bounds_ptr[i_spline * n_queries * para_dim];
1276 }
1277
1278 // eval
1279 core_patches_[i_spline]->SplinepyEvaluate(
1280 &queries[i_query * para_dim],
1281 &sub_patch_centers_ptr[(i_spline * n_queries + i_query) * dim]);
1282 }
1283 };
1284
1285 splinepy::utils::NThreadExecution(calc_sub_patch_centers_step,
1286 n_total,
1287 n_default_threads_);
1288
1289 return sub_patch_centers_;

1472py::array_t<double> PyMultipatch::Sample(const int resolution,
1473 const int nthreads,
1474 const bool same_parametric_bounds) {
1475 const int para_dim = ParaDim();
1476 const int dim = Dim();
1477
1478 // n_queries, and n_splines;
1479 int n_queries{1};
1480 const int n_splines = core_patches_.size();
1481
1482 // prepare resolutions
1483 IntVector resolutions_vector(para_dim);
1484 int* resolutions = resolutions_vector.data();
1485 for (int i{}; i < para_dim; ++i) {
1486 resolutions[i] = resolution;
1487 n_queries *= resolution;
1488 }
1489
1490 // prepare input / output
1491 const int n_total = n_splines * n_queries;
1492 py::array_t<double> sampled({n_total, dim});
1493 double* sampled_ptr = static_cast<double*>(sampled.request().ptr);
1494
1495 // if you know all the queries have same parametric bounds
1496 // you don't need to re-compute queries
1497 if (same_parametric_bounds) {
1498
1499 // get para bounds
1500 DoubleVector para_bounds_vector(2 * para_dim);
1501 double* para_bounds = para_bounds_vector.data();
1502 core_patches_[0]->SplinepyParametricBounds(para_bounds);
1503
1504 // prepare queries
1505 DoubleVector queries_vector(n_queries * para_dim);
1506 double* queries = queries_vector.data();
1507
1508 // use grid point generator to fill queries
1509 splinepy::utils::GridPoints gp_generator(para_dim,
1510 para_bounds,
1511 resolutions);
1512 gp_generator.Fill(queries);
1513
1514 // create lambda for nthread exe
1515 auto sample_same_bounds_step = [&](int, int, const int i_thread) {
1516 for (int i{i_thread}; i < n_total; i += nthreads) {
1517 const auto [i_spline, i_query] = std::div(i, n_queries);
1518 core_patches_[i_spline]->SplinepyEvaluate(
1519 &queries[i_query * para_dim],
1520 &sampled_ptr[(i_spline * n_queries + i_query) * dim]);
1521 }
1522 };
1523
1524 splinepy::utils::NThreadExecution(sample_same_bounds_step,
1525 n_total,
1526 nthreads);
1527
1528 } else {
1529 // here, we will execute 2 times:
1530 // first, to create grid point helpers for each spline
1531 // second, to sample
1532
1533 // create a container to hold grid point helper.
1534 splinepy::utils::DefaultInitializationVector<splinepy::utils::GridPoints>
1535 grid_points(n_splines);
1536
1537 // create grid_points
1538 auto create_grid_points = [&](const int begin, const int end, int) {
1539 DoubleVector para_bounds_vector(2 * para_dim);
1540 double* para_bounds = para_bounds_vector.data();
1541
1542 for (int i{begin}; i < end; ++i) {
1543 // get para_bounds
1544 core_patches_[i]->SplinepyParametricBounds(para_bounds);
1545 // setup grid points helper
1546 grid_points[i].SetUp(para_dim, para_bounds, resolutions);
1547 }
1548 };
1549
1550 // pre compute entries -> this one is a chunk query
1551 splinepy::utils::NThreadExecution(create_grid_points, n_splines, nthreads);
1552
1553 // similar to the one with same_parametric_bounds, except it computes
1554 // query on the fly
1555 auto sample_step = [&](int, int, const int i_thread) {
1556 // each thread needs just one query array
1557 DoubleVector thread_query_vector(para_dim);
1558 double* thread_query = thread_query_vector.data();
1559
1560 for (int i{i_thread}; i < n_total; i += nthreads) {
1561 const auto [i_spline, i_query] = std::div(i, n_queries);
1562 const auto& gp_helper = grid_points[i_spline];
1563 gp_helper.IdToGridPoint(i_query, thread_query);
1564 core_patches_[i_spline]->SplinepyEvaluate(
1565 thread_query,
1566 &sampled_ptr[(i_spline * n_queries + i_query) * dim]);
1567 }
1568 };
1569
1570 // exe - this one is step
1571 splinepy::utils::NThreadExecution(sample_step, n_total, nthreads);
1572 }
1573
1574 return sampled;
1575}

1577void PyMultipatch::AddFields(py::list& fields,
1578 const int field_dim,
1579 const bool check_name,
1580 const bool check_dims,
1581 const bool check_degrees,
1582 const bool check_control_mesh_resolutions,
1583 const int nthreads) {
1584
1585 // allocate space
1586 const auto n_current_fields = static_cast<int>(field_multipatches_.size());
1587 const auto n_new_fields = static_cast<int>(fields.size());
1588
1589 // Reference to original interfaces (required for extraction)
1590 const py::array_t<int> geometric_interfaces = GetInterfaces(false);
1591
1592 // some hint values for null splines
1593 const int para_dim = ParaDim();
1594
1595 // prepare error message
1596 std::string field_mismatch_info{};
1597 std::mutex field_mismatch_mutex;
1598
1599 std::vector<std::shared_ptr<PyMultipatch>> field_ptrs(n_new_fields);
1600
1601 // Multi-threading where each threads deals with a field at a time
1602 auto field_to_multipatch = [&](const int begin, const int end, int) {
1603 // Chunks over field ids
1604 for (int i{begin}; i < end; ++i) {
1605 // create multipatch
1606 py::list casted_list = fields[i].template cast<py::list>();
1607
1608 field_ptrs[i] =
1609 std::make_shared<PyMultipatch>(casted_list, 1, para_dim, field_dim);
1610 field_ptrs[i]->SetInterfaces(geometric_interfaces);
1611
1612 // propagate same_parametric_bounds_ flag
1613 field_ptrs[i]->same_parametric_bounds_ = same_parametric_bounds_;
1614
1615 try {
1616 // check mismatch - doesn't check null splines
1617 RaiseMismatch(core_patches_,
1618 field_ptrs[i]->core_patches_,
1619 check_name,
1620 check_dims,
1621 field_dim,
1622 check_degrees,
1623 check_control_mesh_resolutions,
1624 1);
1625 } catch (const std::runtime_error& e) {
1626 // set true, and add error message.
1627 std::lock_guard<std::mutex> guard(field_mismatch_mutex);
1628 field_mismatch_info += "[mismatch error from the field with index ("
1629 + std::to_string(i) + ")]\n" + e.what();
1630 }
1631 }
1632 };
1633
1634 // Execute
1635 splinepy::utils::NThreadExecution(field_to_multipatch,
1636 n_new_fields,
1637 nthreads);
1638
1639 // raise, if there were error.
1640 if (field_mismatch_info.size() != 0) {
1641 splinepy::utils::PrintAndThrowError(field_mismatch_info);
1642 }
1643
1644 py::list local_fields(n_new_fields);
1645
1646 // turn into a list first to avoid possible runtime memory re-allocation
1647 // caused by `append()`
1648 for (int i{}; i < n_new_fields; ++i) {
1649 local_fields[i] = ToDerived(field_ptrs[i]);
1650 }
1651
1652 // all good, extend to list
1653 field_multipatches_ += local_fields;
