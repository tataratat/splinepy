35template<typename SplineType, typename QueryType, typename OutputType>
36void ScalarTypeEvaluate(const SplineType& spline,
37 const QueryType* query,
38 OutputType* output) {
39 using Query = typename SplineType::ParametricCoordinate_;
40 using QueryValueType = typename Query::value_type;
41 Query core_query;
42
43 // form query
44 for (std::size_t i{}; i < SplineType::kParaDim; ++i) {
45 core_query[i] = QueryValueType{query[i]};
46 }
47
48 // query
49 const auto core_evaluated = spline(core_query);
50
51 // fill output
52 if constexpr (std::is_scalar<decltype(core_evaluated)>::value) {
53 output[0] = static_cast<OutputType>(core_evaluated);
54 } else {
55 for (std::size_t i{}; i < SplineType::kDim; ++i) {
56 output[i] = static_cast<OutputType>(core_evaluated[i]);
57 }
58 }
59}

66template<typename SplineType, typename OutputType>
67void ScalarTypeBoundaryCenters(const SplineType& spline, OutputType* output) {
68 using DoubleVector = splinepy::utils::DefaultInitializationVector<double>;
69
70 // Prepare inputs
71 const int para_dim = spline.SplinepyParaDim();
72
73 // get parametric bounds
74 // They are given back in the order [min_0, min_1,...,max_0, max_1...]
75 DoubleVector bounds_vector(2 * para_dim);
76 double* bounds = bounds_vector.data();
77 spline.SplinepyParametricBounds(bounds);
78
79 // Set parametric coordinate queries
80 for (int i{}; i < para_dim; ++i) {
81 for (int j{}; j < para_dim; ++j) {
82 if (i == j) {
83 output[2 * i * para_dim + j] = bounds[j];
84 output[(2 * i + 1) * para_dim + j] = bounds[j + para_dim];
85 } else {
86 const auto q = .5 * (bounds[j] + bounds[j + para_dim]);
87 output[2 * i * para_dim + j] = q;
88 output[(2 * i + 1) * para_dim + j] = q;
89 }
90 }
91 }
92}

96template<typename SplineType, typename OutputType>
97void ScalarTypeEvaluateBoundaryCenters(const SplineType& spline,
98 OutputType* output) {
99
100 using DoubleVector = splinepy::utils::DefaultInitializationVector<double>;
101
102 // Prepare inputs
103 const int para_dim = spline.SplinepyParaDim();
104 const int dim = spline.SplinepyDim();
105 const int n_faces = 2 * para_dim;
106
107 DoubleVector queries_vector(n_faces * para_dim);
108 double* queries = queries_vector.data();
109
110 // compute queries
111 ScalarTypeBoundaryCenters(spline, queries);
112
113 // Evaluate
114 for (int i{}; i < n_faces; ++i) {
115 spline.SplinepyEvaluate(&queries[i * para_dim], &output[i * dim]);
116 }
117}

164template<typename SplineType, typename QueryType, typename OutputType>
165void ScalarTypeJacobian(const SplineType& spline,
166 const QueryType* query,
167 OutputType* output) {
168 using RealArray = splinepy::utils::Array<double>;
169 using IntArray = splinepy::utils::Array<int>;
170 using RealArray2D = splinepy::utils::Array<double, 2>;
171
172 const int dim = spline.SplinepyDim();
173 // create derivative query, temp result holder, view on output
174 IntArray der_query(SplineType::kParaDim);
175 der_query.Fill(0);
176 RealArray der_result(dim);
177 RealArray2D output_view(output, dim, SplineType::kParaDim);
178
179 // para_dim loop
180 for (int i{}; i < SplineType::kParaDim; ++i) {
181 // prepare eye query
182 ++der_query[i];
183
184 // derivative query
185 spline.SplinepyDerivative(query, der_query.data(), der_result.data());
186
187 // transposed fill
188 for (int j{}; j < dim; ++j) {
189 output_view(j, i) = der_result[j];
190 }
191
192 // reset eye query to zero
193 --der_query[i];
194 }
195}

236template<typename SplineType,
237 typename QueryDimType,
238 typename QueryType,
239 typename IntType>
240int ScalarTypeInsertKnot(SplineType& spline,
241 const QueryDimType query_dim,
242 const QueryType query,
243 const IntType multiplicity) {
244
245 using Dim = typename SplineType::Dimension_;
246 using Knot = typename SplineType::Knot_;
247
248 // since BSpline::InsertKnot is void func, we count knots before and after to
249 // see if it was successful
250 const auto& knot_vector =
251 *spline.GetParameterSpace().GetKnotVectors()[query_dim];
252
253 // Check if request is valid before BSplineLib
254 // see discussions in PR #297
255 // BSplineLib now also has runtime bound check, which raises error.
256 const auto para_bounds = GetParametricBounds(spline);
257 if ((para_bounds[0][query_dim] >= query)
258 || (para_bounds[1][query_dim] <= query)) {
259 return 0;
260 }
261
262 const auto n_knots_before = knot_vector.GetSize();
263
264 spline.InsertKnot(Dim{query_dim}, Knot{query}, multiplicity);
265
266 const auto n_knots_after = knot_vector.GetSize();
267
268 return (n_knots_after - n_knots_before);
269}
