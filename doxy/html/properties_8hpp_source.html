34template<typename SplineType>
35inline std::array<std::array<double, SplineType::kParaDim>, 2>
36GetParametricBounds(const SplineType& spline) {
37 std::array<std::array<double, SplineType::kParaDim>, 2> parametric_bounds{};
38
39 if constexpr (!SplineType::kHasKnotVectors) {
40 parametric_bounds[0].fill(0.);
41 parametric_bounds[1].fill(1.);
42 } else {
43 const auto& parameter_space = spline.GetParameterSpace();
44 const auto& degrees = parameter_space.GetDegrees();
45
46 size_t i{};
47 for (const auto& knotvector : parameter_space.GetKnotVectors()) {
48 const int& degree_value = degrees[i];
49 parametric_bounds[0][i] = knotvector->operator[](degree_value);
50 parametric_bounds[1][i] =
51 knotvector->operator[](knotvector->GetSize() - degree_value - 1);
52 ++i;
53 }
54 }
55
56 return parametric_bounds;
57}

59template<typename SplineType>
60inline int GetNumberOfSupports(const SplineType& spline) {
61 const auto& degrees = spline.GetDegrees();
62 int n_supports{1};
63 for (int i{}; i < SplineType::kParaDim; ++i) {
64 n_supports *= static_cast<int>(degrees[i]) + 1;
65 }
66 return n_supports;
67}

69template<typename ResolutionType = int, typename SplineType>
70inline std::array<ResolutionType, SplineType::kParaDim>
71GetControlMeshResolutions(const SplineType& spline) {
72 std::array<ResolutionType, SplineType::kParaDim> control_mesh_res;
73 const auto& degrees = spline.GetDegrees();
74
75 if constexpr (SplineType::kHasKnotVectors) {
76 const auto& knot_vectors = spline.GetKnotVectors();
77 for (int i{}; i < SplineType::kParaDim; ++i) {
78 control_mesh_res[i] =
79 static_cast<ResolutionType>(knot_vectors[i]->GetSize())
80 - static_cast<ResolutionType>(degrees[i]) - 1;
81 }
82 } else {
83 for (int i{}; i < SplineType::kParaDim; ++i) {
84 control_mesh_res[i] = static_cast<ResolutionType>(degrees[i]) + 1;
85 }
86 }
87
88 return control_mesh_res;
89}

103template<typename SplineType>
104inline void GetGrevilleAbscissae(const SplineType& spline,
105 double* greville_abscissae,
106 const int& i_para_dim,
107 const double& duplicate_tolerance) {
108 // Precompute values
109 const auto& degrees = spline.GetDegrees();
110 // Determine size using control point resolution
111 const int cmr = [&]() {
112 if constexpr (SplineType::kHasKnotVectors) {
113 const auto& knot_vectors = spline.GetKnotVectors();
114 return static_cast<int>(knot_vectors[i_para_dim]->GetSize())
115 - static_cast<int>(degrees[i_para_dim]) - 1;
116
117 } else {
118 return static_cast<int>(degrees[i_para_dim]) + 1;
119 }
120 }();
121
122 // Fill vector with points
123 const double inv_factor = 1. / static_cast<double>(degrees[i_para_dim]);
124 for (int j{}; j < cmr; ++j) {
125 if constexpr (SplineType::kHasKnotVectors) {
126 //
127 const auto& knot_vectors = spline.GetKnotVectors()[i_para_dim];
128 double factor{};
129 for (int k{0}; k < degrees[i_para_dim]; ++k) {
130 factor +=
131 knot_vectors->operator[](typename bsplinelib::Index(k + j + 1));
132 }
133 greville_abscissae[j] = static_cast<double>(inv_factor * factor);
134 } else {
135 greville_abscissae[j] = static_cast<double>(inv_factor * j);
136 }
137 }
138
139 // There can be no duplicates for bezier types, but if C^(-1), duplicates in
140 // the knot vector result in duplicate greville abscissae which can lead to
141 // problems in further computations, solved by mean filtering with neighbors
142 if constexpr (SplineType::kHasKnotVectors) {
143 if (duplicate_tolerance > 0.0) {
144 double previous_knot{greville_abscissae[1]};
145 for (int j{2}; j < cmr - 1; ++j) {
146 if (std::abs(previous_knot - greville_abscissae[j])
147 < duplicate_tolerance) {
148 // @todo make a dynamic tolerance
149 greville_abscissae[j - 1] =
150 0.5 * (greville_abscissae[j - 2] + greville_abscissae[j - 1]);
151 greville_abscissae[j] =
152 0.5 * (greville_abscissae[j] + greville_abscissae[j + 1]);
153 } else {
154 previous_knot = greville_abscissae[j];
155 }
156 }
157 }
158 }
159}
