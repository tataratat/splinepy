39template<int para_dim>
40std::shared_ptr<PySpline>
41ConvertToPySpline(std::shared_ptr<bsplinelib::splines::SplineItem>& spline) {
42 // rational -> NURBS
43 if (spline->is_rational_) {
44 auto bsplinelib_nurbs =
45 bsplinelib::input_output::operations::CastToSpline<para_dim, true>(
46 spline);
47 auto splinepy_nurbs =
48 std::make_shared<splinepy::splines::Nurbs<para_dim>>();
49 // with proper default init, we can just take members as they are shared
50 splinepy_nurbs->ShareMembers(bsplinelib_nurbs);
51
52 return std::make_shared<PySpline>(splinepy_nurbs);
53 }
54
55 auto bsplinelib_bspline =
56 bsplinelib::input_output::operations::CastToSpline<para_dim, false>(
57 spline);
58 auto splinepy_bspline =
59 std::make_shared<splinepy::splines::BSpline<para_dim>>();
60 splinepy_bspline->ShareMembers(bsplinelib_bspline);
61
62 // else -> BSpline
63 return std::make_shared<PySpline>(splinepy_bspline);
64}

66std::shared_ptr<PySpline>
67ToPySpline(std::shared_ptr<bsplinelib::splines::SplineItem>& spline) {
68 switch (spline->parametric_dimensionality_) {
69 case 1:
70 return ConvertToPySpline<1>(spline);
71 case 2:
72 return ConvertToPySpline<2>(spline);
73 default:
74 splinepy::utils::PrintAndThrowError(
75 "invalid para_dim (",
76 spline->parametric_dimensionality_,
77 ") detected while loading spline. It should be less then 3.");
78 break;
79 }
80 return std::shared_ptr<PySpline>{};
81}

83py::list ReadIges(const std::string fname) {
84 // read
85 py::list splines;
86
87 // load
88 auto c_splines = bsplinelib::input_output::iges::Read(fname);
89
90 // convert
91 for (std::shared_ptr<bsplinelib::splines::SplineItem>& spline : c_splines) {
92 splines.append(ToPySpline(spline)->ToDerived());
93 }
94
95 return splines;
96}
