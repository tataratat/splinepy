64 void SetUp(const int dim, const double* bounds, const int* resolutions) {
65 dim_ = dim;
66 len_ = 1;
67 // linspace and prepare possible entries */
68 entries_.resize(dim);
69 resolutions_.resize(dim);
70 step_size_.resize(dim);
71 for (int i{}; i < dim; ++i) {
72 const int& res = resolutions[i];
73 if (res < 2) {
74 splinepy::utils::PrintAndThrowError(
75 "Resolutions for GridPoints can't be less than "
76 "2.");
77 }
78 len_ *= res;
79 resolutions_[i] = res;
80
81 Vector_<double>& entryvec = entries_[i];
82 entryvec.resize(res);
83
84 double& step_size = step_size_[i];
85 // get current lower and upper bounds
86 const double& lower = bounds[i];
87 const double& upper = bounds[i + dim];
88 // get step size
89 step_size = (upper - lower) / static_cast<double>(res - 1);
90 // linspace
91 for (int j{}; j < res; ++j) {
92 entryvec[j] = lower + (step_size * j);
93 }
94 }
95 }

100 void IdToGridPoint(const int& id, double* grid_point) const {
101 int tmp{id};
102 for (int i{0}; i < dim_; ++i) {
103 grid_point[i] = entries_[i][tmp % resolutions_[i]];
104 tmp /= resolutions_[i];
105 }
106 }
107
110 void Fill(double* grid_point_to_fill) const {
111 int i{}, tile{len_}, repeat{1};
112 for (const auto& entry : entries_) {
113 const int entry_size = entry.size();
114 tile /= entry_size;
115 int output_id{i};
116 for (int j{}; j < tile; ++j) {
117 for (const auto& e : entry) {
118 for (int k{}; k < repeat; ++k) {
119 grid_point_to_fill[output_id] = e;
120 output_id += dim_;
121 }
122 }
123 }
124 ++i;
125 repeat *= entry_size;
126 }
127 }

134 static Vector_<int> IdsOnHyperPlane(const int* grid_resolutions,
135 const int& dim,
136 const int& plane_normal_axis,
137 const int& plane_id) {
138 // Determine size of return vector
139 int n_points_on_plane{1};
140 for (int i_pd{}; i_pd < dim; i_pd++) {
141 if (i_pd == plane_normal_axis)
142 continue;
143 n_points_on_plane *= grid_resolutions[i_pd];
144 }
145
146 auto local_to_global_bd_id = [&](const int& local_id) {
147 int combined_offsets{1}, id{local_id}, global_id{};
148 for (int i_pdc{}; i_pdc < dim; ++i_pdc) {
149 const int& res = grid_resolutions[i_pdc];
150 if (i_pdc == plane_normal_axis) {
151 global_id += combined_offsets * plane_id;
152 } else {
153 // Determine index in coordinate indexing system
154 const int rasterized_index = id % res;
155 // Use the current id to update the global index
156 global_id += rasterized_index * combined_offsets;
157 // Update the copy of the index to search for next slice
158 id -= rasterized_index;
159 id /= res;
160 }
161 combined_offsets *= res;
162 }
163 return global_id;
164 };
165
166 // Fill return list
167 Vector_<int> return_list(n_points_on_plane);
168 for (int i{}; i < n_points_on_plane; ++i) {
169 return_list[i] = local_to_global_bd_id(i);
170 }
171 return return_list;
172 }
